### 1.2 核心数据模型

```python
# engine/models/unified_task.py

from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Optional, Dict, List, Any

class TaskPhase(Enum):
    """任务执行阶段（现代公司治理版）"""
    INTAKE = "intake"           # 任务接收
    DRAFT = "draft"             # 方案起草
    REVIEW = "review"            # 评审审议
    DISPATCH = "dispatch"        # 调度派发
    EXECUTE = "execute"         # 部门执行
    QA = "qa"                   # 质量审查
    DONE = "done"               # 完成

    # 终态
    CANCELLED = "cancelled"
    FAILED = "failed"
    TIMEOUT = "timeout"

class Department(Enum):
    """部门枚举"""
    HR = "hr"                   # 人力资源部
    FINANCE = "finance"          # 财务部
    ADMIN = "admin"              # 行政部
    OPERATIONS = "operations"     # 运营部
    LEGAL = "legal"              # 法务部
    ENGINEERING = "engineering"   # 工程部

@dataclass
class FlowEntry:
    """流转日志条目"""
    from_phase: Optional[TaskPhase]
    to_phase: TaskPhase
    reason: str
    triggered_by: str
    timestamp: datetime = field(default_factory=datetime.now)
    metadata: Dict[str, Any] = field(default_factory=dict)

@dataclass
class DepartmentResult:
    """部门执行结果"""
    department: Department
    output: str
    status: str  # "success" / "failed" / "blocked"
    timestamp: datetime = field(default_factory=datetime.now)

@dataclass
class UnifiedTask:
    """
    统一任务上下文

    贯穿整个任务生命周期，包含：
    - 执行状态（状态机）
    - 讨论流程（COO workflow）
    - 流转日志（flow_log）
    - 部门执行结果
    """
    task_id: str
    trace_id: str

    # 身份信息
    title: str = ""
    description: str = ""
    user_input: str = ""
    priority: str = "normal"

    # 执行状态
    phase: TaskPhase = TaskPhase.INTAKE
    target_department: Optional[Department] = None

    # COO 工作流
    workflow_phase: str = "layer1"  # "layer1" / "layer2" / "layer3" / "layer4" / "completed"

    # 流转日志
    flow_log: List[FlowEntry] = field(default_factory=list)
    progress_log: List[str] = field(default_factory=list)

    # 部门执行
    department_results: Dict[Department, DepartmentResult] = field(default_factory=dict)

    # 质量评分
    quality_score: Optional[float] = None

    # 时间戳
    created_at: datetime = field(default_factory=datetime.now)
    updated_at: datetime = field(default_factory=datetime.now)
    completed_at: Optional[datetime] = None

    def add_flow(self, from_phase: Optional[TaskPhase], to_phase: TaskPhase,
                 reason: str, triggered_by: str, metadata: dict = None):
        """添加流转日志"""
        entry = FlowEntry(
            from_phase=from_phase,
            to_phase=to_phase,
            reason=reason,
            triggered_by=triggered_by,
            metadata=metadata or {}
        )
        self.flow_log.append(entry)
        self.updated_at = datetime.now()

    def transition_to(self, new_phase: TaskPhase, reason: str, triggered_by: str):
        """状态转换"""
        if not self.can_transition_to(new_phase):
            raise InvalidTransitionError(
                f"Cannot transition from {self.phase.value} to {new_phase.value}"
            )
        self.add_flow(self.phase, new_phase, reason, triggered_by)
        self.phase = new_phase

    def can_transition_to(self, target: TaskPhase) -> bool:
        """检查是否可转换"""
        allowed = TRANSITION_RULES.get(self.phase, set())
        return target in allowed

    def is_terminal(self) -> bool:
        """是否终态"""
        return self.phase in {TaskPhase.DONE, TaskPhase.CANCELLED, TaskPhase.FAILED, TaskPhase.TIMEOUT}
```

### 1.3 状态转换规则

```python
# 合法的状态转换
TRANSITION_RULES = {
    TaskPhase.INTAKE: {TaskPhase.DRAFT, TaskPhase.CANCELLED},
    TaskPhase.DRAFT: {TaskPhase.REVIEW, TaskPhase.CANCELLED},
    TaskPhase.REVIEW: {TaskPhase.DISPATCH, TaskPhase.DRAFT, TaskPhase.CANCELLED},
    TaskPhase.DISPATCH: {TaskPhase.EXECUTE, TaskPhase.CANCELLED},
    TaskPhase.EXECUTE: {TaskPhase.QA, TaskPhase.FAILED, TaskPhase.TIMEOUT, TaskPhase.CANCELLED},
    TaskPhase.QA: {TaskPhase.DONE, TaskPhase.EXECUTE, TaskPhase.CANCELLED},
    TaskPhase.DONE: set(),  # 终态
    TaskPhase.CANCELLED: set(),  # 终态
    TaskPhase.FAILED: set(),  # 终态
    TaskPhase.TIMEOUT: set(),  # 终态
}
```
