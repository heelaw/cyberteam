## 一、融合背景与目标

### 1.1 为什么融合

| 痛点 | 参考项目解法 |
|------|-------------|
| SSE 流无法跨 HMR 保持状态 | CodePilot `globalThis` 单例模式 |
| 多模型 Provider 切换混乱 | CodePilot `provider-resolver` 抽象 + 凭证隔离 |
| 工具发现依赖硬编码 | Magic `ToolFactory` 自动扫描 + 延迟加载 |
| 任务无持久化追踪 | Magic `TaskSchedulerDomainService` DDD 架构 |
| 专家 Agent 无注册/发现机制 | Magic 数字员工市场模式 |
| AI 传错参数无友好提示 | Magic 苏格拉底式追问 + 相似度匹配 |

### 1.2 融合目标

```
Phase 1 (当前)：前端流管理增强 + 工具工厂模式
  → SSE 全局单例（跨 HMR）
  → ToolFactory 自动发现
  → Pydantic 参数验证 + 友好错误

Phase 2 (短期)：任务调度系统 + 数字员工市场
  → TaskScheduler 持久化
  → Expert Agent 注册/发现
  → Human-in-the-loop 审批门控

Phase 3 (中期)：Agent 协作增强
  → 审批门控 + 成本控制
  → 个人秘书层（用户代理）
```

## 二、P0 融合：前端 SSE 流管理（CodePilot 模式）

### 2.1 现状问题

当前 CyberTeam 的 SSE 订阅依赖 React 组件生命周期：
- 组件卸载 → EventSource 断开
- 组件重新挂载 → 从头开始，无快照
- 无法跨路由共享流状态

### 2.2 融合方案：StreamSessionManager

参考 `CodePilot/src/lib/stream-session-manager.ts`，创建 `frontend/src/lib/stream-session-manager.ts`：

```typescript
// 全局单例，跨越 HMR 和路由切换
type StreamState = 'connecting' | 'active' | 'idle' | 'closed'

interface StreamSession {
  id: string
  taskId: string
  state: StreamState
  lastActivity: number
  abortController: AbortController
  emitter: EventEmitter
  // 快照：最新接收到的消息
  snapshot: SSEStreamEvent[]
}

class StreamSessionManager {
  private static _instance: StreamSessionManager
  sessions: Map<string, StreamSession>

  // 启动新流
  startStream(taskId: string, message: string): StreamSession

  // 订阅已有流（立即获取快照 + 实时更新）
  subscribe(sessionId: string, callbacks: SSEStreamCallbacks): () => void

  // 空闲超时（5min）+ GC（每5min清理）
  // Idle 超时自动中断
  // GC 清理超过5min的已关闭会话
}
```

        # 2. 从环境变量/配置解析
        # 3. 回退到默认 Provider

        # 返回：max_tokens, supports_streaming, supports_vision...

# 值对象
class TaskStatus(Enum):
    UNKNOWN = 0
    PENDING = 1      # 创建/等待
    RUNNING = 2      # 执行中
    SUCCESS = 3      # 成功完成
    FAILED = 4       # 失败
    CANCELLED = 5    # 取消
    TIMEOUT = 6      # 超时
    RETRY = 7        # 重试中

# 实体
class TaskScheduler:
    id: str
    name: str
    task_type: TaskType  # IMMEDIATE=1, SCHEDULED=2, CRON=3
    status: TaskStatus
    expect_time: datetime
    actual_time: Optional[datetime]
    retry_times: int
    max_retries: int
    callback_method: str
    callback_params: dict
    result: Optional[TaskResult]

# 领域服务
class TaskSchedulerDomainService:
    async def create_task(self, name, task_type, callback, params):
        """创建即时任务"""

    async def create_scheduled_task(self, name, run_at, callback, params):
        """创建定时任务"""

    async def create_cron_task(self, name, cron_expr, callback, params):
        """创建 Cron 任务"""

    async def execute(self, task_id: str):
        """执行任务（含重试逻辑）"""

    async def cancel(self, task_id: str):
        """取消任务"""

# 事件
class TaskEvent(Enum):
    TASK_CREATED = "task.created"
    TASK_STARTED = "task.started"
    TASK_COMPLETED = "task.completed"
    TASK_FAILED = "task.failed"

```python
# 高危操作审批门控
class ApprovalLevel(Enum):
    AUTO = 0        # 自动通过
    LOW = 1        # 部门总监审批
    MEDIUM = 2     # COO 审批
    HIGH = 3       # CEO 审批

class ApprovalGate:
    def __init__(self):
        self._rules: list[ApprovalRule] = []

    def register_rule(self, action_pattern: str, level: ApprovalLevel):
        """注册审批规则"""

    async def request_approval(self, action: str, context: dict) -> str:
        """发起审批请求，返回 approval_id"""

    async def approve(self, approval_id: str, approver: str):
        """审批通过"""

    async def reject(self, approval_id: str, reason: str):
        """审批拒绝"""

# 预置高危动作
APPROVAL_RULES = [
    ("DELETE.*", ApprovalLevel.MEDIUM),    # 删除操作
    ("EXECUTE_SQL.*", ApprovalLevel.HIGH), # SQL 执行
    ("SEND_EXTERNAL.*", ApprovalLevel.MEDIUM), # 外部发送
    ("MODIFY_BUDGET.*", ApprovalLevel.HIGH),  # 预算修改
]
```
