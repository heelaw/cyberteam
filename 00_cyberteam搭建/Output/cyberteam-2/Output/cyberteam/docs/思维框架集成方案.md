### 触发规则

```python
def trigger_framework(user_input):
    """根据输入自动触发合适的思维框架"""

    # 关键词匹配
    keywords = extract_keywords(user_input)

    # 框架推荐
    frameworks = []
    for kw in keywords:
        matched = match_framework(kw)
        frameworks.extend(matched)

    # 优先级排序
    ranked = rank_frameworks(frameworks)

    return ranked[:3]  # 返回前3个推荐框架
```

### 使用示例

```python
# 1. 问题输入
user_input = "如何提升用户留存率？"

# 2. 框架推荐
engine = ThinkingFrameworkEngine()
recommended = engine.recommend({
    "keywords": ["用户", "留存", "提升"],
    "intent": "增长"
})
# -> [AARRR, 同期群分析, RFM模型]

# 3. 多框架应用

    solution = engine.apply(fw, user_input)
    solutions.append(solution)

# 4. 结果聚合
final = engine.aggregate(solutions)
```

## 四、与 CEO Agent 集成

```python
class CEOAgent:
    def __init__(self):
        self.framework_engine = ThinkingFrameworkEngine()

    def analyze(self, user_input):
        # 获取框架推荐
        frameworks = self.framework_engine.recommend({
            "keywords": extract_keywords(user_input),
            "intent": classify_intent(user_input)
        })

        # 应用多个框架
        solutions = []
        for fw in frameworks:
            solution = self.framework_engine.apply(fw, user_input)
            solutions.append(solution)

        # 聚合结果
        final_solution = self.framework_engine.aggregate(solutions)

        return {
            "frameworks_used": [f.name for f in frameworks],
            "solution": final_solution
        }
```
