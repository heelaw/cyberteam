## 二、Agent 协作流程

### 任务选择逻辑

```
用户需求 → 需求分类 → Agent匹配 → 任务分发 → 执行 → 结果聚合
```

### Agent 选择矩阵

| 需求类型 | 主 Agent | 辅助 Agent | 输出 |
|----------|----------|------------|------|
| 用户增长 | growth-agent | activity-agent, incentive-agent | 增长方案 |
| 内容营销 | newmedia-agent | copywriter-agent | 内容计划 |
| 活动策划 | activity-agent | planning-agent | 活动方案 |
| 会员体系 | incentive-agent | framework-agent | 激励方案 |
| 战略规划 | strategy-agent | planning-agent | 战略文档 |
| 投资分析 | investor-agent | strategy-agent | 投资报告 |

### 协作模式

```python
class Agent协作:
    def __init__(self):
        self.agents = load_agents()

    def select_agents(self, task_type):
        """根据任务类型选择合适的 Agent"""

        selection_map = {
            "增长": {
                "primary": "growth-agent",
                "secondary": ["activity-agent", "incentive-agent"],
                "tertiary": ["data-agent"]
            },
            "内容": {
                "primary": "newmedia-agent",
                "secondary": ["copywriter-agent"],
                "tertiary": ["planning-agent"]
            },
            "活动": {
                "primary": "activity-agent",
                "secondary": ["planning-agent"],
                "tertiary": ["newmedia-agent"]
            },
            "会员": {
                "primary": "incentive-agent",
                "secondary": ["framework-agent"],
                "tertiary": ["data-agent"]
            },
            "战略": {
                "primary": "strategy-agent",
                "secondary": ["planning-agent"],
                "tertiary": ["investor-agent"]
            },
            "投资": {
                "primary": "investor-agent",
                "secondary": ["strategy-agent"],
                "tertiary": ["data-agent"]
            }
        }

        return selection_map.get(task_type, {
            "primary": "general-agent",
            "secondary": [],
            "tertiary": []
        })

    def collaborate(self, task_spec):
        """多 Agent 协作执行"""

        # 选择 Agent
        selection = self.select_agents(task_spec["type"])

        # 启动主 Agent
        primary_result = self.run_agent(
            selection["primary"],
            task_spec
        )

        # 并行启动辅助 Agent
        secondary_results = []
        for agent in selection["secondary"]:
            result = self.run_agent(agent, task_spec)
            secondary_results.append(result)

        # 聚合结果
        return self.aggregate(
            primary_result,
            secondary_results,
            selection
        )

    def aggregate(self, primary, secondary, selection):
        """聚合多 Agent 结果"""

        return {
            "main": primary,
            "support": secondary,
            "agents_used": [
                selection["primary"]
            ] + selection["secondary"],
            "final_solution": self.merge_solutions(
                primary,
                secondary
            )
        }
```

## 五、与 CEO Agent 集成

### 调用接口

```python
class CyberTeamCEO:
    def __init__(self):
        self.agent_collaborator = Agent协作()

    def process_task(self, user_input):
        """处理用户任务"""

        # 需求分类
        task_type = self.classify(user_input)

        # 选择合适的 Agent 组合
        selection = self.agent_collaborator.select_agents(task_type)

        # 执行协作
        result = self.agent_collaborator.collaborate({
            "type": task_type,
            "input": user_input,
            "agents": selection
        })

        return result
```
