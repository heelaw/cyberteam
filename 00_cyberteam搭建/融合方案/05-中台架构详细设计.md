### 2.5 专家匹配算法

#### 2.5.1 匹配算法核心

```python
class ExpertMatchingAlgorithm:
    """
    专家匹配算法 - 基于多维度相似度计算
    """

    def match(self, task: Task, available_experts: List[Expert]) -> List[ExpertRecommendation]:
        """
        匹配流程:
        1. 任务特征提取
        2. 专家能力向量化
        3. 多维度相似度计算
        4. 协同性优化
        5. 最终排序输出
        """

        # Step 1: 任务特征提取
        task_features = self.extract_task_features(task)

        # Step 2: 专家能力向量构建
        expert_vectors = [self.build_expert_vector(e) for e in available_experts]

        # Step 3: 多维度相似度计算
        similarity_scores = []
        for expert, vector in zip(available_experts, expert_vectors):
            scores = {
                "semantic_similarity": self.cosine_similarity(task_features, vector),
                "domain_match": self.domain_similarity(task.domain, expert.domain),
                "scenario_relevance": self.scenario_relevance(task.scenario, expert.scenarios),
                "historical_performance": self.get_historical_score(task.type, expert.id)
            }

            # Step 4: 加权综合评分
            final_score = (
                scores["semantic_similarity"] * 0.35 +
                scores["domain_match"] * 0.25 +
                scores["scenario_relevance"] * 0.25 +
                scores["historical_performance"] * 0.15
            )

            similarity_scores.append((expert, scores, final_score))

        # Step 5: 协同性优化 (避免相似专家重复注入)
        optimized_list = self.synergy_optimization(similarity_scores)

        return optimized_list

    def synergy_optimization(self, candidates: List) -> List:
        """
        协同性优化:
        - 避免同类型专家重复注入
        - 优先选择互补性强的专家组合
        """
        # 按最终评分排序
        sorted_candidates = sorted(candidates, key=lambda x: x[2], reverse=True)

        # 选择互补的专家组合
        selected = []
        selected_categories = set()

        for candidate in sorted_candidates:
            category = candidate[0].category
            if category not in selected_categories:
                selected.append(candidate)
                selected_categories.add(category)

        return selected
```

#### 3.3.2 资产发现机制

```python
class AssetDiscovery:
    """
    资产发现引擎 - 多维度检索
    """

    def discover(self, query: DiscoveryQuery) -> List[Asset]:
        """
        发现策略:
        1. 关键词精确匹配
        2. 语义相似度搜索
        3. 能力图谱推理
        4. 使用历史推荐
        """

        # 多策略并行检索
        results = []

        # 策略1: 关键词匹配
        keyword_results = self.keyword_search(query)
        results.extend(keyword_results)

        # 策略2: 语义搜索 (向量相似度)
        semantic_results = self.semantic_search(query)
        results.extend(semantic_results)

        # 策略3: 能力图谱推理
        graph_results = self.capability_graph_search(query)

class HandoffProtocol:
    """
    Agent间Handoff协议实现
    """

    def initiate_handoff(self, task: Task, from_agent: str, to_agent: str,
                        handoff_type: str, context: HandoffContext) -> HandoffResult:
        """
        Handoff流程:
        1. 上下文捕获
        2. 完整性校验
        3. 安全传输
        4. 接收确认
        5. 结果回写
        """

        # Step 1: 上下文捕获
        captured_context = {
            "task_state": task.get_state(),
            "intermediate_outputs": self.capture_intermediates(task),
            "execution_history": self.get_execution_history(task),
            "knowledge_artifacts": self.capture_knowledge(task),
            "pending_dependencies": self.get_pending_deps(task),
            "quality_metrics": self.get_current_metrics(task)
        }

        # Step 2: 完整性校验
        integrity_check = self.verify_context_integrity(captured_context, handoff_type)
        if not integrity_check.is_complete:
            # 触发补全流程
            self.request_context_completion(integrity_check.missing_parts)

        # Step 3: 传输加密与签名
        encrypted_context = self.encrypt_and_sign(captured_context)

        # Step 4: 发送并等待确认
        confirmation = await self.send_with_confirmation(
            to_agent,
            MessageType.HANDOFF_TRANSFER,
            encrypted_context
        )

        # Step 5: 结果回写
        if confirmation.status == "accepted":
            task.update_owner(to_agent)
            task.add_handoff_record(from_agent, to_agent, handoff_type)
            return HandoffResult.SUCCESS
        else:
            return HandoffResult.RETRY_OR_ESCALATE
```

### 5.5 记忆检索机制

#### 5.5.1 混合检索策略

```python
class HybridMemoryRetrieval:
    """
    混合记忆检索 - 结合向量检索和关键词检索
    """

    def retrieve(self, query: Query, top_k: int = 10) -> List[MemoryResult]:
        """
        检索流程:
        1. 向量相似度检索
        2. 关键词全文检索
        3. BM25排序
        4. RRF融合
        5. 重排序 (Rerank)
        """

        results = []

        # 并行执行多种检索
        vector_results = self.vector_search(query, top_k * 2)
        keyword_results = self.keyword_search(query, top_k * 2)
        bm25_results = self.bm25_search(query, top_k * 2)

        # RRF融合 (Reciprocal Rank Fusion)
        fused_scores = self.rrf_fusion([
            vector_results,
            keyword_results,
            bm25_results
        ])

        # 重排序
        reranked = self.rerank(query, fused_results)

        return reranked[:top_k]

    def rrf_fusion(self, result_lists: List[List], k: int = 60) -> Dict:
        """
        RRF融合算法
        """
        scores = defaultdict(float)

        for result_list in result_lists:
            for rank, item in enumerate(result_list):
                item_id = item.id
                # RRF公式: 1 / (k + rank)
                scores[item_id] += 1 / (k + rank)

        return sorted(scores.items(), key=lambda x: x[1], reverse=True)
```

```python
class LoopDetection:
    """
    循环检测引擎
    """

    def detect_execution_loop(self, task: Task) -> LoopDetectionResult:
        """
        执行循环检测:
        基于状态机历史检测循环模式
        """

        history = task.get_state_history()

        # 检测模式: PENDING -> RUNNING -> PENDING
        transitions = self.extract_transitions(history)

        # 统计状态循环
        pending_count = sum(1 for t in transitions if t.to_state == "PENDING")
        failed_count = task.failure_count

        if pending_count > self.thresholds["execution_loop"]:
            return LoopDetectionResult(
                detected=True,
                loop_type="execution",
                severity="high",
                details={
                    "pending_transitions": pending_count,
                    "failure_count": failed_count,
                    "recommendation": "retry_limit_or_escalate"
                }
            )

        return LoopDetectionResult(detected=False)

    def detect_message_loop(self, mailbox: MailboxManager) -> LoopDetectionResult:
        """
        通信循环检测:
        基于消息传递图检测死循环
        """

        message_graph = mailbox.get_message_graph()

        # 检测环 (Cycle Detection via DFS)
        cycles = self.find_cycles(message_graph)

        if cycles:
            return LoopDetectionResult(
                detected=True,
                loop_type="message",
                severity="medium",
                details={
                    "cycles_found": len(cycles),
                    "cycle_paths": cycles[:3],  # 最多返回3个
                    "recommendation": "break_cycle_via_timeout"
                }
            )

        return LoopDetectionResult(detected=False)
```

class TaskProgress:
    """任务进度模型"""

    def __init__(self, task_id: str):
        self.task_id = task_id
        self.total_steps: int = 0
        self.completed_steps: int = 0
        self.current_step: str = ""
        self.start_time: datetime = None
        self.expected_end_time: datetime = None
        self.actual_end_time: datetime = None
        self.blockers: List[str] = []
        self.progress_percentage: float = 0.0

    def calculate_progress(self) -> float:
        """
        进度计算:
        - 简单任务: completed_steps / total_steps
        - 复杂任务: 加权里程碑计算
        """

        if self.total_steps == 0:
            return 0.0

        # 基础进度
        base_progress = (self.completed_steps / self.total_steps) * 100

        # 里程碑加权
        milestone_bonus = self.calculate_milestone_bonus()

        self.progress_percentage = min(base_progress + milestone_bonus, 99.9)
        return self.progress_percentage

    def get_eta(self) -> Optional[datetime]:
        """预计完成时间"""
        if self.completed_steps == 0:
            return None

        elapsed = datetime.now() - self.start_time
        avg_step_time = elapsed / self.completed_steps
        remaining_steps = self.total_steps - self.completed_steps

        return datetime.now() + (avg_step_time * remaining_steps)
```

class ExperienceRecommender:
    """
    经验推荐引擎
    基于任务上下文推荐相关经验
    """

    def recommend(self, task: Task, top_k: int = 5) -> List[Experience]:
        """
        推荐流程:
        1. 任务特征提取
        2. 经验匹配 (向量检索)
        3. 场景过滤
        4. 效果排序
        """

        # 任务特征
        task_features = self.extract_features(task)

        # 向量检索
        candidates = self.vector_store.search(
            query_vector=task_features,
            top_k=top_k * 2
        )

        # 场景过滤
        filtered = [
            exp for exp in candidates
            if self.is_context_matched(task, exp)
        ]

        # 效果排序
        ranked = sorted(
            filtered,
            key=lambda x: (x.effectiveness, x.usage_count),
            reverse=True
        )

        return ranked[:top_k]

    def is_context_matched(self, task: Task, experience: Experience) -> bool:
        """
        场景匹配判断:
        - 任务类型匹配
        - 行业场景匹配
        - 复杂度匹配
        - 资源约束匹配
        """
        # 任务类型
        if task.type not in experience.applicable_types:
            return False

        # 复杂度适配
        complexity_diff = abs(task.complexity - experience.typical_complexity)
        if complexity_diff > 2:
            return False

        # 时效性检查
        if experience.is_outdated():
            return False

        return True
```
