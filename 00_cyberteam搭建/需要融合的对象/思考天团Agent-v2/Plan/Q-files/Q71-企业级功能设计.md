## 二、身份认证 (SSO)

### 2.1 SSO架构

```python
# auth/sso_manager.py
from typing import Dict, List, Optional
from dataclasses import dataclass
from enum import Enum

class AuthProvider(Enum):
    NONE = "none"                    # 无认证
    OAUTH2 = "oauth2"               # OAuth2.0
    SAML = "saml"                   # SAML
    LDAP = "ldap"                   # LDAP/AD
    OIDC = "oidc"                   # OpenID Connect

@dataclass
class User:
    id: str
    email: str
    name: str
    role: str
    department: str = ""
    auth_provider: str = ""

@dataclass
class SSOAuthConfig:
    provider: AuthProvider
    client_id: str
    client_secret: str
    authorization_url: str
    token_url: str
    userinfo_url: str
    scopes: List[str]

class SSOManager:
    """SSO认证管理器"""

    def __init__(self):
        self.providers: Dict[AuthProvider, SSOAuthConfig] = {}

    def configure(self, config: SSOAuthConfig):
        """配置认证提供者"""
        self.providers[config.provider] = config

    def authenticate(self, provider: AuthProvider, code: str) -> User:
        """OAuth2认证流程"""
        config = self.providers.get(provider)
        if not config:
            raise ValueError(f"Provider {provider} not configured")

        # 1. 用code换取token
        token = self._exchange_code(code, config)

        # 2. 用token获取用户信息
        userinfo = self._get_userinfo(token, config)

        # 3. 映射到本地用户
        return User(
            id=userinfo["sub"],
            email=userinfo["email"],
            name=userinfo["name"],
            role=userinfo.get("role", "user"),
            department=userinfo.get("department", ""),
            auth_provider=provider.value
        )

    def _exchange_code(self, code: str, config: SSOAuthConfig) -> Dict:
        """用code换取token"""
        response = requests.post(
            config.token_url,
            # P0修正: OAuth2 RFC 要求 Content-Type: application/json 或 application/x-www-form-urlencoded
            # 原 data={} 发送的是 form-encoded (需 Content-Type: application/x-www-form-urlencoded)
            # 使用 json={} 让 requests 自动设置 Content-Type: application/json
            json={
                "grant_type": "authorization_code",
                "code": code,
                "client_id": config.client_id,
                "client_secret": config.client_secret
            }
        )
        return response.json()

    def _get_userinfo(self, token: Dict, config: SSOAuthConfig) -> Dict:
        """获取用户信息"""
        response = requests.get(
            config.userinfo_url,
            headers={"Authorization": f"Bearer {token['access_token']}"}
        )
        return response.json()
```

```python
# auth/rbac.py
from typing import Dict, List, Set
from dataclasses import dataclass, field
from enum import Enum

class Role(Enum):
    ADMIN = "admin"           # 管理员
    MANAGER = "manager"      # 经理
    ANALYST = "analyst"      # 分析师
    VIEWER = "viewer"        # 查看者

class Permission(Enum):
    # 问题管理
    QUESTION_CREATE = "question:create"
    QUESTION_READ = "question:read"
    QUESTION_UPDATE = "question:update"
    QUESTION_DELETE = "question:delete"

    # 专家管理
    EXPERT_EXECUTE = "expert:execute"
    EXPERT_CONFIGURE = "expert:configure"

    # 报告管理
    REPORT_READ = "report:read"
    REPORT_EXPORT = "report:export"
    REPORT_DELETE = "report:delete"

    # 系统管理
    USER_MANAGE = "user:manage"
    CONFIG_UPDATE = "config:update"
    AUDIT_READ = "audit:read"

# 角色权限映射
ROLE_PERMISSIONS = {
    Role.ADMIN: [
        Permission.QUESTION_CREATE,
        Permission.QUESTION_READ,
        Permission.QUESTION_UPDATE,
        Permission.QUESTION_DELETE,
        Permission.EXPERT_EXECUTE,
        Permission.EXPERT_CONFIGURE,
        Permission.REPORT_READ,
        Permission.REPORT_EXPORT,
        Permission.REPORT_DELETE,
        Permission.USER_MANAGE,
        Permission.CONFIG_UPDATE,
        Permission.AUDIT_READ,
    ],
    Role.MANAGER: [
        Permission.QUESTION_CREATE,
        Permission.QUESTION_READ,
        Permission.QUESTION_UPDATE,
        Permission.EXPERT_EXECUTE,
        Permission.REPORT_READ,
        Permission.REPORT_EXPORT,
        Permission.AUDIT_READ,
    ],
    Role.ANALYST: [
        Permission.QUESTION_CREATE,
        Permission.QUESTION_READ,
        Permission.EXPERT_EXECUTE,
        Permission.REPORT_READ,
    ],
    Role.VIEWER: [
        Permission.QUESTION_READ,
        Permission.REPORT_READ,
    ]
}

@dataclass
class UserContext:
    """用户上下文"""
    user_id: str
    role: Role
    department: str = ""
    permissions: Set[Permission] = field(default_factory=set)

    def __post_init__(self):
        self.permissions = set(ROLE_PERMISSIONS.get(self.role, []))

    def has_permission(self, permission: Permission) -> bool:
        """检查权限"""
        return permission in self.permissions

class RBACManager:
    """RBAC访问控制管理器"""

    def __init__(self):
        self.role_assignments: Dict[str, Role] = {}
        self.department_roles: Dict[str, str] = {}  # department -> role

    def assign_role(self, user_id: str, role: Role):
        """分配角色"""
        self.role_assignments[user_id] = role

    def get_user_context(self, user_id: str, department: str = "") -> UserContext:
        """获取用户上下文"""
        role = self.role_assignments.get(user_id, Role.VIEWER)
        return UserContext(user_id=user_id, role=role, department=department)

    def authorize(self, user_id: str, permission: Permission) -> bool:
        """授权检查"""
        context = self.get_user_context(user_id)
        return context.has_permission(permission)
```

### 6.1 Webhook服务

```python
# webhook/webhook_manager.py
from typing import Dict, List, Callable
from dataclasses import dataclass
from enum import Enum
# P2修正: requests 已统一在文件顶部模块级导入，移除重复导入

class WebhookEvent(Enum):
    QUESTION_CREATED = "question.created"
    QUESTION_COMPLETED = "question.completed"
    QUESTION_FAILED = "question.failed"
    REPORT_GENERATED = "report.generated"
    EXPERT_INVOKED = "expert.invoked"

@dataclass
class WebhookConfig:
    """Webhook配置"""
    url: str
    events: List[WebhookEvent]
    secret: str
    enabled: bool = True
    retry_count: int = 3
    retry_delay: int = 5  # 秒

class WebhookManager:
    """Webhook管理器"""

    def __init__(self):
        self.webhooks: List[WebhookConfig] = []
        self.event_handlers: Dict[WebhookEvent, List[Callable]] = {}

    def register(self, webhook: WebhookConfig):
        """注册Webhook"""
        self.webhooks.append(webhook)

    def unregister(self, url: str):
        """注销Webhook"""
        self.webhooks = [w for w in self.webhooks if w.url != url]

    def trigger(self, event: WebhookEvent, data: Dict):
        """触发Webhook"""
        payload = {
            "event": event.value,
            "timestamp": time.time(),
            "data": data
        }

        for webhook in self.webhooks:
            if not webhook.enabled:
                continue

            if event not in webhook.events:
                continue

            self._send_webhook(webhook, payload)

    def _send_webhook(self, webhook: WebhookConfig, payload: Dict):
        """发送Webhook"""
        # 生成签名
        signature = hmac.new(
            webhook.secret.encode(),
            json.dumps(payload).encode(),
            hashlib.sha256
        ).hexdigest()

        headers = {
            "Content-Type": "application/json",
            "X-Webhook-Signature": signature,
            "X-Webhook-Event": payload["event"]
        }

        # 重试逻辑
        for attempt in range(webhook.retry_count):
            try:
                response = requests.post(
                    webhook.url,
                    json=payload,
                    headers=headers,
                    timeout=10
                )

                if response.status_code == 200:
                    return  # 成功

            except requests.RequestException as e:
                print(f"Webhook attempt {attempt + 1} failed: {e}")

            # 等待后重试
            if attempt < webhook.retry_count - 1:
                time.sleep(webhook.retry_delay)

        # 所有重试都失败
        self._handle_failure(webhook, payload)

    def _handle_failure(self, webhook: WebhookConfig, payload: Dict):
        """处理失败"""
        # 记录失败日志
```
