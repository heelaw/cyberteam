### 2.1 TaskQueue（任务队列）

```python
# task_queue.py
from dataclasses import dataclass, field
from enum import Enum
from typing import List, Optional
import time
import uuid

class TaskPriority(Enum):
    LOW = 0
    NORMAL = 1
    HIGH = 2
    URGENT = 3

@dataclass
class QueuedTask:
    task_id: str = field(default_factory=lambda: str(uuid.uuid4()))
    expert_id: str = ""
    question_id: str = ""
    prompt: str = ""
    priority: TaskPriority = TaskPriority.NORMAL
    dependencies: List[str] = field(default_factory=list)  # 依赖的任务ID
    created_at: float = field(default_factory=time.time)
    status: str = "queued"  # queued, running, completed, failed, cancelled

    def __lt__(self, other):
        # 优先级队列比较
        if self.priority != other.priority:
            return self.priority.value > other.priority.value
        return self.created_at < other.created_at

class TaskQueue:
    def __init__(self, max_size: int = 1000):
        self.max_size = max_size
        self._queue: List[QueuedTask] = []
        self._task_map: dict = {}  # task_id -> QueuedTask

    def enqueue(self, task: QueuedTask) -> bool:
        """入队，返回是否成功"""
        if len(self._queue) >= self.max_size:
            return False
        self._queue.append(task)
        self._task_map[task.task_id] = task
        self._queue.sort(reverse=True)  # 优先级排序
        return True

    def dequeue(self) -> Optional[QueuedTask]:
        """出队（最高优先级）"""
        if not self._queue:
            return None
        task = self._queue.pop(0)
        return task

    def get(self, task_id: str) -> Optional[QueuedTask]:
        """获取任务"""
        return self._task_map.get(task_id)

    def update_status(self, task_id: str, status: str):
        """更新任务状态"""
        if task_id in self._task_map:
            self._task_map[task_id].status = status

    def size(self) -> int:
        return len(self._queue)

    def get_pending(self, expert_id: str = None) -> List[QueuedTask]:
        """获取待执行任务"""
        result = []
        for task in self._queue:
            if task.status == "queued":
                if expert_id is None or task.expert_id == expert_id:
                    result.append(task)
        return result
```

### 2.2 TaskScheduler（任务调度器）

```python
# task_scheduler.py
from typing import List, Dict, Callable, Optional
import asyncio
import time
from concurrent.futures import ThreadPoolExecutor, Future
import logging

logger = logging.getLogger(__name__)

class TaskScheduler:
    def __init__(self, max_concurrent: int = 4):
        self.max_concurrent = max_concurrent
        self._running: Dict[str, Future] = {}  # task_id -> Future
        self._executor = ThreadPoolExecutor(max_workers=max_concurrent)
        self._callbacks: Dict[str, Callable] = {}

    def submit(self, task_id: str, fn: Callable, *args, **kwargs) -> Future:
        """提交任务"""
        # 检查并发限制
        while len(self._running) >= self.max_concurrent:
            time.sleep(0.1)  # 等待

        future = self._executor.submit(fn, *args, **kwargs)
        self._running[task_id] = future

        # 添加完成回调
        def on_complete(f):
            self._running.pop(task_id, None)
            if task_id in self._callbacks:
                self._callbacks[task_id](f)

        future.add_done_callback(on_complete)
        return future

    def submit_async(self, task_id: str, coro):
        """提交异步任务"""
        loop = asyncio.get_event_loop()
        future = asyncio.run_coroutine_threadsafe(coro, loop)
        self._running[task_id] = future

        def on_complete(f):
            self._running.pop(task_id, None)

        future.add_done_callback(on_complete)
        return future

    def cancel(self, task_id: str) -> bool:
        """取消任务"""
        if task_id in self._running:
            cancelled = self._running[task_id].cancel()
            if cancelled:
                self._running.pop(task_id, None)
            return cancelled
        return False

    def wait_all(self, timeout: float = None) -> Dict[str, any]:
        """等待所有任务完成"""
        results = {}
        for task_id, future in self._running.items():
            try:
                results[task_id] = future.result(timeout=timeout)
            except Exception as e:
                results[task_id] = {"error": str(e)}
        return results

    def get_running_count(self) -> int:
        """获取运行中的任务数"""
        return len(self._running)

    def is_full(self) -> bool:
        """是否已满载"""
        return len(self._running) >= self.max_concurrent
```

### 2.3 TimeoutController（超时控制）

```python
# timeout_controller.py
import time
import threading
from typing import Dict, Callable, Optional, Any
from dataclasses import dataclass
import logging

logger = logging.getLogger(__name__)

@dataclass
class TimeoutConfig:
    default_timeout: float = 300  # 默认5分钟
    expert_timeouts: Dict[str, float] = None  # 专家特定超时

    def __post_init__(self):
        if self.expert_timeouts is None:
            self.expert_timeouts = {
                "kahneman": 180,
                "first_principle": 300,
                "six_hats": 240,
                "grow": 180,
                "reverse_thinking": 180,
                "default": 300
            }

    def get_timeout(self, expert_id: str) -> float:
        return self.expert_timeouts.get(expert_id, self.default_timeout)

class TimeoutController:
    def __init__(self, config: TimeoutConfig = None):
        self.config = config or TimeoutConfig()
        self._timers: Dict[str, threading.Timer] = {}
        self._callbacks: Dict[str, Callable] = {}
        self._task_start_times: Dict[str, float] = {}
        self._original_timeouts: Dict[str, float] = {}  # 保存原始超时值

    def start(self, task_id: str, expert_id: str, on_timeout: Callable):
        """启动超时计时器"""
        timeout = self.config.get_timeout(expert_id)
        self._callbacks[task_id] = on_timeout
        self._task_start_times[task_id] = time.time()
        self._original_timeouts[task_id] = timeout  # 保存原始超时值

        timer = threading.Timer(timeout, self._on_timeout, args=(task_id,))
        timer.daemon = True
        self._timers[task_id] = timer
        timer.start()

        logger.info(f"Started timeout timer for {task_id}: {timeout}s")

    def cancel(self, task_id: str):
        """取消超时计时"""
        if task_id in self._timers:
            self._timers[task_id].cancel()
            self._timers.pop(task_id, None)
            self._callbacks.pop(task_id, None)
            self._task_start_times.pop(task_id, None)
            self._original_timeouts.pop(task_id, None)

    def extend(self, task_id: str, extra_time: float):
        """延长超时时间"""
        if task_id in self._timers:
            # 取消当前计时器
            self._timers[task_id].cancel()

            # 计算剩余时间
            elapsed = time.time() - self._task_start_times.get(task_id, 0)
            # 使用保存的原始超时值计算剩余时间
            original_timeout = self._original_timeouts.get(task_id, 300)
            remaining = original_timeout - elapsed

            # 创建新的计时器
            new_timeout = max(remaining + extra_time, 60)  # 至少60秒
            timer = threading.Timer(new_timeout, self._on_timeout, args=(task_id,))
            timer.daemon = True
            self._timers[task_id] = timer
            # 更新保存的超时值
            self._original_timeouts[task_id] = new_timeout
            timer.start()

    def _on_timeout(self, task_id: str):
        """超时触发"""
        logger.warning(f"Task {task_id} timed out!")
        if task_id in self._callbacks:
            try:
                self._callbacks[task_id](task_id)
            except Exception as e:
                logger.error(f"Timeout callback error: {e}")

        # 清理
        self._timers.pop(task_id, None)
        self._callbacks.pop(task_id, None)

    def get_elapsed(self, task_id: str) -> float:
        """获取已用时间"""
        start = self._task_start_times.get(task_id)
        if start is None:
            return 0
        return time.time() - start

    # P0修复: 添加remaining()方法
    def remaining(self, task_id: str) -> float:
        """获取剩余时间（秒）

        P0修复: 使用 timer.interval - elapsed 计算剩余时间，
        而非错误地使用 is_alive() 返回bool
        """
        timer = self._timers.get(task_id)
        if timer is None:
            return 0
        # timer.interval 是设定的超时时间
        elapsed = self.get_elapsed(task_id)
        return max(0, timer.interval - elapsed)
```

### 2.4 ResultCollector（结果收集器）

```python
# result_collector.py
from typing import Dict, List, Optional, Any
from dataclasses import dataclass, field
from enum import Enum
import time
import threading
import json
import logging

logger = logging.getLogger(__name__)

class ResultStatus(Enum):
    PENDING = "pending"
    RUNNING = "running"
    COMPLETED = "completed"
    FAILED = "failed"
    TIMEOUT = "timeout"
    CANCELLED = "cancelled"

@dataclass
class TaskResult:
    task_id: str
    expert_id: str
    status: ResultStatus = ResultStatus.PENDING
    output: Any = None
    error: Optional[str] = None
    started_at: Optional[float] = None
    completed_at: Optional[float] = None
    duration: Optional[float] = None
    tokens_used: int = 0
    metadata: Dict = field(default_factory=dict)

    def to_dict(self) -> dict:
        return {
            "task_id": self.task_id,
            "expert_id": self.expert_id,
            "status": self.status.value,
            "output": self.output,
            "error": self.error,
            "started_at": self.started_at,
            "completed_at": self.completed_at,
            "duration": self.duration,
            "tokens_used": self.tokens_used,
            "metadata": self.metadata
        }

class ResultCollector:
    def __init__(self, question_id: str, output_dir: str):
        self.question_id = question_id
        self.output_dir = output_dir
        self._results: Dict[str, TaskResult] = {}
        self._lock = threading.Lock()
        self._completed_event = threading.Event()

    def start(self, task_id: str, expert_id: str) -> TaskResult:
        """记录任务开始"""
        with self._lock:
            result = TaskResult(
                task_id=task_id,
                expert_id=expert_id,
                status=ResultStatus.RUNNING,
                started_at=time.time()
            )
            self._results[task_id] = result
            self._save_to_file(result)
            return result

    def complete(self, task_id: str, output: Any, tokens_used: int = 0, metadata: Dict = None):
        """记录任务完成"""
        with self._lock:
            if task_id not in self._results:
                logger.warning(f"Task {task_id} not found in results")
                return

            result = self._results[task_id]
            result.status = ResultStatus.COMPLETED
            result.output = output
            result.completed_at = time.time()
            result.duration = result.completed_at - result.started_at
            result.tokens_used = tokens_used
            if metadata:
                result.metadata.update(metadata)

            self._save_to_file(result)
            self._check_completion()

    def fail(self, task_id: str, error: str):
        """记录任务失败"""
        with self._lock:
            if task_id not in self._results:
                return

            result = self._results[task_id]
            result.status = ResultStatus.FAILED
            result.error = error
            result.completed_at = time.time()
            result.duration = result.completed_at - result.started_at

            self._save_to_file(result)
            self._check_completion()

    def timeout(self, task_id: str):
        """记录任务超时"""
        self.fail(task_id, "Task timeout")

    def get(self, task_id: str) -> Optional[TaskResult]:
        """获取结果"""
        return self._results.get(task_id)

    def get_all(self) -> List[TaskResult]:
        """获取所有结果"""
        return list(self._results.values())

    def get_completed(self) -> List[TaskResult]:
        """获取已完成的结果"""
        return [r for r in self._results.values() if r.status == ResultStatus.COMPLETED]

    def get_failed(self) -> List[TaskResult]:
        """获取失败的结果"""
        return [r for r in self._results.values() if r.status == ResultStatus.FAILED]

    def is_all_completed(self) -> bool:
        """是否全部完成"""
        if not self._results:
            return False
        return all(r.status == ResultStatus.COMPLETED for r in self._results.values())

    def wait_until_complete(self, timeout: float = None) -> bool:
        """等待全部完成"""
        return self._completed_event.wait(timeout)

    def _check_completion(self):
        """检查是否全部完成"""
        if self.is_all_completed():
            self._completed_event.set()

    def _save_to_file(self, result: TaskResult):
        """保存到文件"""
        import os
        os.makedirs(self.output_dir, exist_ok=True)
        file_path = os.path.join(self.output_dir, f"{result.task_id}.json")
        with open(file_path, 'w') as f:
            json.dump(result.to_dict(), f, ensure_ascii=False, indent=2)

    def export_summary(self) -> dict:
        """导出汇总"""
        completed = self.get_completed()
        failed = self.get_failed()

        return {
            "question_id": self.question_id,
            "total": len(self._results),
            "completed": len(completed),
            "failed": len(failed),
            "success_rate": len(completed) / len(self._results) if self._results else 0,
            "total_duration": sum(r.duration for r in completed if r.duration),
            "total_tokens": sum(r.tokens_used for r in completed),
            "results": [r.to_dict() for r in self._results.values()]
        }
```

### 3.1 ParallelExecutor（并行执行器）

```python
# parallel_executor.py
from typing import List, Dict, Callable, Optional
import logging
from concurrent.futures import ThreadPoolExecutor, as_completed

from task_queue import TaskQueue, QueuedTask, TaskPriority
from task_scheduler import TaskScheduler
from timeout_controller import TimeoutController, TimeoutConfig
from result_collector import ResultCollector, ResultStatus

logger = logging.getLogger(__name__)

class ParallelExecutor:
    def __init__(
        self,
        max_concurrent: int = 4,
        timeout_config: TimeoutConfig = None,
        output_dir: str = "thinking-team/workspace/tasks"
    ):
        self.max_concurrent = max_concurrent
        self.timeout_config = timeout_config or TimeoutConfig()

        self.task_queue = TaskQueue()
        self.scheduler = TaskScheduler(max_concurrent=max_concurrent)
        self.timeout_controller = TimeoutController(self.timeout_config)

        self._running_tasks: Dict[str, str] = {}  # task_id -> question_id

    def execute_parallel(
        self,
        question_id: str,
        tasks: List[Dict],
        execute_fn: Callable,  # 执行函数: (expert_id, prompt) -> result
        on_task_complete: Callable = None
    ) -> Dict:
        """
        并行执行多个专家任务

        Args:
            question_id: 问题ID
            tasks: 任务列表 [{"expert_id": "kahneman", "prompt": "..."}, ...]
            execute_fn: 执行函数
            on_task_complete: 任务完成回调

        Returns:
            执行结果汇总
        """
        output_dir = f"thinking-team/workspace/tasks/{question_id}/results"
        collector = ResultCollector(question_id, output_dir)

        # 1. 入队所有任务
        for task in tasks:
            queued_task = QueuedTask(
                expert_id=task["expert_id"],
                question_id=question_id,
                prompt=task["prompt"],
                priority=TaskPriority(task.get("priority", 1))
            )
            self.task_queue.enqueue(queued_task)

        # 2. 并行执行
        with ThreadPoolExecutor(max_workers=self.max_concurrent) as executor:
            futures = {}

            for task in tasks:
                task_id = f"{question_id}_{task['expert_id']}"
                future = executor.submit(
                    self._execute_single_task,
                    task_id,
                    task["expert_id"],
                    task["prompt"],
                    collector,
                    execute_fn
                )
                futures[task_id] = future

            # 3. 等待完成
            for future in as_completed(futures):
                task_id = futures[future]
                try:
                    result = future.result()
                    if on_task_complete:
                        on_task_complete(task_id, result)
                except Exception as e:
                    logger.error(f"Task {task_id} failed: {e}")
                    collector.fail(task_id, str(e))

        # 4. 返回汇总
        return collector.export_summary()

    def _execute_single_task(
        self,
        task_id: str,
        expert_id: str,
        prompt: str,
        collector: ResultCollector,
        execute_fn: Callable
    ):
        """执行单个任务"""
        # 记录开始
        collector.start(task_id, expert_id)

        # 启动超时计时器
        self.timeout_controller.start(
            task_id,
            expert_id,
            lambda tid: collector.timeout(tid)
        )

        try:
            # 执行
            result = execute_fn(expert_id, prompt)

            # 记录完成
            collector.complete(
                task_id,
                output=result.get("output"),
                tokens_used=result.get("tokens_used", 0),
                metadata=result.get("metadata", {})
            )

            # 取消超时计时器
            self.timeout_controller.cancel(task_id)

            return result

        except Exception as e:
            logger.error(f"Task {task_id} error: {e}")
            collector.fail(task_id, str(e))
            self.timeout_controller.cancel(task_id)
            raise

    def execute_chain(
        self,
        question_id: str,
        tasks: List[Dict],
        execute_fn: Callable,
        combine_fn: Callable = None
    ) -> Dict:
        """
        链式执行（上一个的输出作为下一个的输入）

        Args:
            question_id: 问题ID
            tasks: 任务列表（按顺序）
            execute_fn: 执行函数
            combine_fn: 组合函数（可选）

        Returns:
            执行结果
        """
        output_dir = f"thinking-team/workspace/tasks/{question_id}/results"
        collector = ResultCollector(question_id, output_dir)

        context = {}

        for i, task in enumerate(tasks):
            task_id = f"{question_id}_chain_{i}"

            # 如果有前一个任务的输出，合并到prompt
            if context and "context" in task:
                task["prompt"] = combine_fn(
                    task["prompt"],
                    context
                ) if combine_fn else task["prompt"]

            # 执行
            collector.start(task_id, task["expert_id"])

            try:
                result = execute_fn(task["expert_id"], task["prompt"])

                collector.complete(
                    task_id,
                    output=result.get("output"),
                    tokens_used=result.get("tokens_used", 0)
                )

                # 保存上下文
                context[task["expert_id"]] = result.get("output")

            except Exception as e:
                logger.error(f"Chain task {task_id} failed: {e}")
                collector.fail(task_id, str(e))
                break

        return collector.export_summary()

    def execute_debate(
        self,
        question_id: str,
        tasks: List[Dict],
        execute_fn: Callable,
        max_rounds: int = 3
    ) -> Dict:
        """
        辩论模式执行（多轮交互）

        Args:
            question_id: 问题ID
            tasks: 初始任务列表
            execute_fn: 执行函数
            max_rounds: 最大轮数

        Returns:
            辩论结果
        """
        output_dir = f"thinking-team/workspace/tasks/{question_id}/results"
        collector = ResultCollector(question_id, output_dir)

        # 第一轮：并行执行初始分析
        expert_outputs = {}
        for task in tasks:
            task_id = f"{question_id}_round0_{task['expert_id']}"
            collector.start(task_id, task["expert_id"])

            try:
                result = execute_fn(task["expert_id"], task["prompt"])
                collector.complete(task_id, output=result.get("output"))
                expert_outputs[task["expert_id"]] = result.get("output")
            except Exception as e:
                collector.fail(task_id, str(e))

        # 后续轮：基于他人输出进行辩论
        for round_num in range(1, max_rounds):
            new_outputs = {}

            for task in tasks:
                # 构建包含他人观点的prompt
                debate_prompt = self._build_debate_prompt(
                    task["prompt"],
                    expert_outputs,
                    task["expert_id"]
                )

                task_id = f"{question_id}_round{round_num}_{task['expert_id']}"
                collector.start(task_id, task["expert_id"])

                try:
                    result = execute_fn(task["expert_id"], debate_prompt)
                    collector.complete(task_id, output=result.get("output"))
                    new_outputs[task["expert_id"]] = result.get("output")
                except Exception as e:
                    collector.fail(task_id, str(e))

            # 检查是否收敛（所有专家达成共识）
            if self._check_convergence(new_outputs, expert_outputs):
                logger.info(f"Debate converged at round {round_num}")
                break

            expert_outputs = new_outputs

        return collector.export_summary()

    def _build_debate_prompt(self, original_prompt: str, other_outputs: Dict, my_expert_id: str) -> str:
        """构建辩论prompt"""
        others_view = "\n\n".join([
            f"### {expert_id}的观点:\n{output}"
            for expert_id, output in other_outputs.items()
            if expert_id != my_expert_id
        ])

        return f"""
原始问题: {original_prompt}

其他专家的观点:
{others_view}

请基于以上观点，提出你的反驳或补充意见。
"""

    def _check_convergence(self, new_outputs: Dict, old_outputs: Dict) -> bool:
        """检查是否收敛

        P1修正: 修复维度错误 - 原来是 changed <= len(...) * 0.3
        (count vs count×percentage 比较无意义)
        修正为: 变化比例 = changed / len(...) <= 0.3
        """

        if not old_outputs:
            return False

        # 统计变化的专家数量
        changed = 0
        total = len(new_outputs)

        for expert_id in new_outputs:
            if expert_id in old_outputs:
                if new_outputs[expert_id] != old_outputs[expert_id]:
                    changed += 1

        # P1修正: 比较变化比例 vs 阈值
        if total == 0:
            return True

        change_ratio = changed / total
        return change_ratio <= 0.3
```
