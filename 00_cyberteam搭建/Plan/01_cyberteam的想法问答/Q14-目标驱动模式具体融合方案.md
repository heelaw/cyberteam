### 4.2 实现代码（直接可用）

```python
class GoalDrivenExecutor:
    """目标驱动执行器 - 直接可用"""

    def __init__(self, goal, criteria):
        self.goal = goal
        self.criteria = criteria  # 必须是可量化函数
        self.iteration = 0
        self.best_result = None

    async def execute(self):
        """执行目标驱动循环"""

        while not self.is_done() and self.iteration < self.config["max_iterations"]:
            self.iteration += 1

            # 1. 执行一轮
            result = await self.run_one_iteration()

            # 2. 评估结果
            improved = self.evaluate(result)

            if improved:
                self.best_result = result
                log(f"迭代 {self.iteration}: 改进！")
            else:
                log(f"迭代 {self.iteration}: 无改进")

            # 3. 检查成功标准
            if self.check_success_criteria(result):
                log("成功标准满足，停止")
                return result

            # 4. PUA监督（可选）
            if self.pua_supervisor:
                feedback = await self.pua_supervisor.supervise(result)
                if feedback.needs_retry:
                    continue

        return self.best_result

    def is_done(self):
        """检查是否应该停止"""
        # 时间检查
        if self.elapsed_time > self.config["max_time_seconds"]:
            return True

        # 花费检查
        if self.total_cost > self.config["max_cost_usd"]:
            return True

        # 用户手动停止
        if self.user_requested_stop:
            return True

        return False
```

### 5.1 在执行层添加"目标驱动循环"

```python
# 你的执行层代码
class ExecutionLayer:
    def __init__(self):
        self.agents = []           # 执行Agent
        self.pua_supervisor = PUASupervisor()
        self.goal_driven = None   # 新增：目标驱动器

    async def execute(self, approved_plan):
        """
        执行用户批准的计划
        """
        # 方案1：直接执行（当前）
        # return await self.simple_execute(approved_plan)

        # 方案2：目标驱动执行（新增）
        self.goal_driven = GoalDrivenExecutor(
            goal=approved_plan.goal,
            criteria=approved_plan.success_criteria
        )

        return await self.goal_driven.execute()
```
