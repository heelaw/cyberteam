```python
# input_validator.py
from typing import Dict, List, Optional
from dataclasses import dataclass
from enum import Enum

class ValidationLevel(Enum):
    PASS = "pass"
    WARNING = "warning"
    FAIL = "fail"

@dataclass
class ValidationResult:
    level: ValidationLevel
    message: str
    details: List[str] = None

class InputValidator:
    """L0: 输入校验"""

    # 问题最小长度
    MIN_QUESTION_LENGTH = 10
    # 问题最大长度
    MAX_QUESTION_LENGTH = 2000

    # 必需的上下文标签
    REQUIRED_TAGS = []  # 可选，根据业务需求

    def validate(self, question: str, context: Dict = None) -> ValidationResult:
        """校验输入"""
        errors = []
        warnings = []

        # 1. 长度检查
        if len(question) < self.MIN_QUESTION_LENGTH:
            errors.append(f"问题太短（最少{self.MIN_QUESTION_LENGTH}字符）")

        if len(question) > self.MAX_QUESTION_LENGTH:
            errors.append(f"问题太长（最多{self.MAX_QUESTION_LENGTH}字符）")

        # 2. 内容检查
        if question.strip() == "":
            errors.append("问题不能为空")

        # 3. 敏感词检查
        sensitive_words = self._check_sensitive(question)
        if sensitive_words:
            warnings.append(f"包含敏感词: {sensitive_words}")

        # 4. 重复问题检查
        if self._is_duplicate(question):
            warnings.append("可能与历史问题重复")

        # 5. 复杂度评估
        complexity = self._assess_complexity(question)
        if complexity > 8:
            warnings.append("问题复杂度高，可能需要更多专家")

        # 返回结果
        if errors:
            return ValidationResult(
                level=ValidationLevel.FAIL,
                message="输入校验失败",
                details=errors
            )

        if warnings:
            return ValidationResult(
                level=ValidationLevel.WARNING,
                message="输入需注意",
                details=warnings
            )

        return ValidationResult(
            level=ValidationLevel.PASS,
            message="输入校验通过"
        )

    def _check_sensitive(self, text: str) -> List[str]:
        """检查敏感词"""
        sensitive_words = ["政治", "色情", "暴力", "诈骗"]  # 示例
        found = [w for w in sensitive_words if w in text]
        return found

    def _is_duplicate(self, question: str) -> bool:
        """检查重复"""
        # 简单实现：可以用文本相似度
        return False

    def _assess_complexity(self, question: str) -> float:
        """评估复杂度"""
        score = 0

        # 关键词计数
        complex_keywords = ["如何", "为什么", "策略", "分析", "评估", "优化"]
        score += sum(1 for kw in complex_keywords if kw in question)

        # 长度
        score += len(question) / 500

        # 问号数量（多个问题）
        score += question.count("？") + question.count("?") - 1

        return min(score, 10)
```

### 2.2 L1: 计划审批

```python
# plan_approver.py
from typing import Dict, List
from dataclasses import dataclass
from enum import Enum

class ApprovalStatus(Enum):
    APPROVED = "approved"
    REJECTED = "rejected"
    NEEDS_REVISION = "needs_revision"

@dataclass
class PlanApproval:
    status: ApprovalStatus
    score: float  # 0-100
    feedback: str
    suggestions: List[str] = None

class PlanApprover:
    """L1: 计划审批"""

    # 通过阈值
    PASS_SCORE = 70
    # 良好阈值
    GOOD_SCORE = 85

    def approve(self, intent: Dict, experts: List[str], mode: str) -> PlanApproval:
        """审批执行计划"""

        score = 0
        feedback = []
        suggestions = []

        # 1. 意图清晰度 (30分)
        intent_score = self._score_intent(intent)
        score += intent_score
        if intent_score < 20:
            feedback.append("意图识别不够清晰")

        # 2. 专家匹配度 (40分)
        match_score = self._score_expert_match(experts, intent)
        score += match_score
        if match_score < 25:
            feedback.append("专家匹配度不足")
            suggestions.append(f"建议添加: {self._suggest_experts(intent)}")

        # 3. 模式合理性 (20分)
        mode_score = self._score_mode(mode, intent, experts)
        score += mode_score
        if mode_score < 15:
            feedback.append("执行模式选择可能不当")

        # 4. 资源配置 (10分)
        resource_score = self._score_resources(experts)
        score += resource_score

        # 判断状态
        if score >= self.GOOD_SCORE:
            status = ApprovalStatus.APPROVED
            feedback = ["计划质量优秀"] + feedback
        elif score >= self.PASS_SCORE:
            status = ApprovalStatus.APPROVED
        else:
            status = ApprovalStatus.REJECTED

        return PlanApproval(
            status=status,
            score=score,
            feedback="; ".join(feedback) if feedback else "计划通过",
            suggestions=suggestions
        )

    def _score_intent(self, intent: Dict) -> float:
        """评分意图清晰度"""
        score = 0

        if intent.get("primary"):
            score += 15

        if intent.get("keywords"):
            score += min(len(intent["keywords"]) * 3, 15)

        return score

    def _score_expert_match(self, experts: List[str], intent: Dict) -> float:
        """评分专家匹配度"""
        # 简单实现：检查是否匹配意图类型
        score = 30 if experts else 0

        # 根据问题类型检查
        intent_type = intent.get("primary", "")
        type_experts = {
            "决策": ["kahneman", "six_hats"],
            "创新": ["first_principle", "reverse_thinking"],
            "诊断": ["five_why", "reverse_thinking"],
            "战略": ["swot_tows", "mckinsey"],
            "目标": ["grow", "goldlin"],
            "执行": ["wbs", "grow"],
            "管理": ["manager_leap"],
            "技术": ["ai_board", "first_principle"]
        }

        matched = sum(1 for e in experts if e in type_experts.get(intent_type, []))
        score = min(matched * 10 + 10, 40)

        return score

    def _score_mode(self, mode: str, intent: Dict, experts: List[str]) -> float:
        """评分模式合理性"""
        score = 10  # 默认基础分

        if mode == "parallel" and len(experts) >= 3:
            score += 10
        elif mode == "debate" and len(experts) >= 2:
            score += 10
        elif mode == "chain" and len(experts) >= 2:
            score += 5  # 链式不需要太多专家

        return min(score, 20)

    def _score_resources(self, experts: List[str]) -> float:
        """评分资源配置"""
        # 专家数量
        count = len(experts)
        if count < 2:
            return 3
        elif count <= 4:
            return 10
        elif count <= 6:
            return 8
        else:
            return 5  # 太多专家可能效率低

    def _suggest_experts(self, intent: Dict) -> List[str]:
        """建议专家"""
        # 根据意图类型建议
        return []
```

### 2.3 L2: 过程检查

```python
# process_checker.py
from typing import Dict, List, Optional
from dataclasses import dataclass
from enum import Enum
import time

class ProcessStatus(Enum):
    NORMAL = "normal"
    WARNING = "warning"
    SLOW = "slow"
    BLOCKED = "blocked"
    FAILED = "failed"

@dataclass
class ProcessCheck:
    status: ProcessStatus
    progress: float  # 0-100
    elapsed_time: float
    estimated_remaining: float
    alerts: List[str] = None

class ProcessChecker:
    """L2: 过程检查"""

    # 超时阈值（秒）
    EXPERT_TIMEOUT = 300
    # 慢速阈值（秒）
    SLOW_THRESHOLD = 180
    # 正常进度百分比
    NORMAL_PROGRESS = 50

    def check(self, task_states: Dict[str, Dict]) -> ProcessCheck:
        """检查过程状态"""

        total = len(task_states)
        completed = sum(1 for s in task_states.values() if s.get("status") == "completed")
        failed = sum(1 for s in task_states.values() if s.get("status") == "failed")
        running = sum(1 for s in task_states.values() if s.get("status") == "running")

        progress = (completed / total * 100) if total > 0 else 0
        elapsed = self._get_elapsed_time(task_states)

        alerts = []

        # 检查失败
        if failed > 0:
            status = ProcessStatus.FAILED
            alerts.append(f"{failed}个任务失败")

        # 检查阻塞
        elif running == 0 and completed < total:
            status = ProcessStatus.BLOCKED
            alerts.append("任务阻塞，无运行中的任务")

        # 检查慢速
        elif elapsed > self.SLOW_THRESHOLD and progress < self.NORMAL_PROGRESS:
            status = ProcessStatus.SLOW
            alerts.append("任务执行缓慢")

        # 检查警告
        elif progress < self.NORMAL_PROGRESS * (elapsed / self.SLOW_THRESHOLD):
            status = ProcessStatus.WARNING
            alerts.append("进度落后于预期")

        else:
            status = ProcessStatus.NORMAL

        # 估算剩余时间
        if completed > 0:
            avg_time = elapsed / completed
            estimated_remaining = avg_time * (total - completed)
        else:
            estimated_remaining = self.EXPERT_TIMEOUT

        return ProcessCheck(
            status=status,
            progress=progress,
            elapsed_time=elapsed,
            estimated_remaining=estimated_remaining,
            alerts=alerts if alerts else None
        )

    def _get_elapsed_time(self, task_states: Dict) -> float:
        """获取经过时间"""
        start_times = [
            s.get("started_at", 0)
            for s in task_states.values()
            if s.get("started_at")
        ]

        if not start_times:
            return 0

        earliest = min(start_times)
        return time.time() - earliest
```

### 2.4 L3: 结果评审

```python
# result_reviewer.py
# P2修复: L3 调用 Q70 UnifiedQualityScorer，实现六维度统一评分
from typing import Dict, List
from dataclasses import dataclass
from enum import Enum

# 导入 Q70 统一评分器
import sys
import os
sys.path.insert(0, os.path.dirname(os.path.dirname(__file__)))
try:
    from Q70_unified_quality_scorer import UnifiedQualityScorer, QualityScore
except ImportError:
    # 内联备用 (如果 Q70 模块不可用)
    from unified_quality_scorer import UnifiedQualityScorer, QualityScore

class ReviewStatus(Enum):
    PASS = "pass"
    NEEDS_REVISION = "needs_revision"
    FAIL = "fail"

@dataclass
class ReviewResult:
    status: ReviewStatus
    score: float  # 0-100
    dimensions: Dict[str, float]  # 各维度评分
    feedback: str
    issues: List[str] = None

class ResultReviewer:
    """L3: 结果评审 (调用 Q70 UnifiedQualityScorer)"""

    # 通过阈值 (与 Q70 一致: C级60分及以上通过)
    PASS_SCORE = 60
    # 修订阈值 (D级45分需修订)
    REVISION_THRESHOLD = 45

    def __init__(self):
        self.scorer = UnifiedQualityScorer()

    def review(self, outputs: Dict[str, Dict]) -> ReviewResult:
        """评审专家输出 (委托给 Q70 UnifiedQualityScorer)"""

        # 合并多个专家输出为一个分析对象
        combined_output = self._combine_outputs(outputs)

        # 调用 Q70 统一评分器 (stage="result_review")
        quality_score: QualityScore = self.scorer.score(
            output=combined_output,
            stage="result_review"
        )

        # 转换为 Q66 ReviewResult 格式
        status = self._map_level_to_status(quality_score.level)

        # 生成反馈
        feedback = self._generate_feedback(quality_score)

        return ReviewResult(
            status=status,
            score=quality_score.total,
            dimensions=quality_score.breakdown,
            feedback=feedback,
            issues=quality_score.issues
        )

    def _combine_outputs(self, outputs: Dict[str, Dict]) -> Dict:
        """合并多个专家输出为一个分析对象"""
        if not outputs:
            return {"output": "", "expert_id": "unknown"}

        combined_content = []
        expert_ids = []

        for expert_id, output in outputs.items():
            expert_ids.append(expert_id)
            content = output.get("output", "")
            if content:
                combined_content.append(f"【{expert_id}】{content}")

        return {
            "output": "\n\n".join(combined_content),
            "expert_ids": expert_ids,
            "expert_id": ",".join(expert_ids)
        }

    def _map_level_to_status(self, level: str) -> ReviewStatus:
        """将 Q70 等级映射到 Q66 状态"""
        level_map = {
            "A": ReviewStatus.PASS,      # 90-100: 优秀直接通过
            "B": ReviewStatus.PASS,      # 75-89: 良好微调后通过
            "C": ReviewStatus.NEEDS_REVISION,  # 60-74: 合格需要修订
            "D": ReviewStatus.NEEDS_REVISION,  # 45-59: 较差需要重做
            "F": ReviewStatus.FAIL       # <45: 不合格拒绝
        }
        return level_map.get(level, ReviewStatus.FAIL)

    def _generate_feedback(self, quality_score: QualityScore) -> str:
        """生成人类可读反馈"""
        feedback_parts = []

        # 等级说明
        level_names = {
            "A": "优秀",
            "B": "良好",
            "C": "合格",
            "D": "较差",
            "F": "不合格"
        }
        feedback_parts.append(f"质量等级: {level_names.get(quality_score.level, '未知')} ({quality_score.level})")

        # 问题维度
        if quality_score.issues:
            feedback_parts.append(f"问题: {', '.join(quality_score.issues)}")

        # 分数说明
        feedback_parts.append(f"总分: {quality_score.total:.1f}/100")

        return " | ".join(feedback_parts)
```

### 2.5 L4: 交付终审

```python
# delivery_approver.py
from typing import Dict, List
from dataclasses import dataclass
from enum import Enum

class DeliveryStatus(Enum):
    APPROVED = "approved"
    NEEDS_REVISION = "needs_revision"
    REJECTED = "rejected"

@dataclass
class DeliveryApproval:
    status: DeliveryStatus
    score: float
    summary_quality: float
    recommendation_quality: float
    risk_level: str
    feedback: str

class DeliveryApprover:
    """L4: 交付终审"""

    PASS_SCORE = 75

    def approve(self, final_report: Dict, expert_outputs: Dict) -> DeliveryApproval:
        """终审交付物"""

        # 1. 报告质量
        summary_score = self._score_summary(final_report.get("summary", ""))
        recommendation_score = self._score_recommendations(final_report.get("recommendations", []))
        report_score = (summary_score + recommendation_score) / 2

        # 2. 共识度
        consensus_score = self._score_consensus(expert_outputs)

        # 3. 风险评估
        risk_level = self._assess_risk(final_report, expert_outputs)

        # 总分
        total_score = (report_score * 0.5 + consensus_score * 0.5)

        # 判断状态
        if total_score >= self.PASS_SCORE and risk_level in ["low", "medium"]:
            status = DeliveryStatus.APPROVED
        elif risk_level == "high":
            status = DeliveryStatus.REJECTED
        else:
            status = DeliveryStatus.NEEDS_REVISION

        return DeliveryApproval(
            status=status,
            score=total_score,
            summary_quality=summary_score,
            recommendation_quality=recommendation_score,
            risk_level=risk_level,
            feedback=self._generate_delivery_feedback(status, report_score, consensus_score)
        )

    def _score_summary(self, summary: str) -> float:
        """评分摘要质量"""
        score = 0

        if len(summary) > 50:
            score += 25

        if "核心" in summary or "关键" in summary:
            score += 25

        if len(summary) < 300:  # 简洁
            score += 25

        if any(kw in summary for kw in ["问题", "分析", "建议"]):
            score += 25

        return min(score, 100)

    def _score_recommendations(self, recommendations: List) -> float:
        """评分建议质量"""
        if not recommendations:
            return 0

        score = 0

        for rec in recommendations:
            # 有具体行动
            if rec.get("action"):
                score += 15

            # 有责任人
            if rec.get("owner"):
                score += 10

            # 有时间线
            if rec.get("timeline"):
                score += 10

        return min(score, 100)

    def _score_consensus(self, expert_outputs: Dict) -> float:
        """评分共识度"""
        if not expert_outputs:
            return 0

        # 简化：检查是否有共识标记
        consensus_count = 0

        for output in expert_outputs.values():
            content = output.get("output", "")
            if "共识" in content or "一致" in content or "同意" in content:
                consensus_count += 1

        return (consensus_count / len(expert_outputs)) * 100

    def _assess_risk(self, final_report: Dict, expert_outputs: Dict) -> str:
        """评估风险"""
        # 简单实现
        risks = final_report.get("risks", [])

        high_risk_count = sum(1 for r in risks if r.get("impact") == "high")

        if high_risk_count >= 3:
            return "high"
        elif high_risk_count >= 1:
            return "medium"
        else:
            return "low"

    def _generate_delivery_feedback(self, status, report_score, consensus_score) -> str:
        """生成反馈"""
        if status == DeliveryStatus.APPROVED:
            return "交付物通过终审"
        elif status == DeliveryStatus.NEEDS_REVISION:
            return f"报告质量{report_score:.0f}分，共识度{consensus_score:.0f}分，需要修订"
        else:
            return "风险过高，拒绝交付"
```

```python
# feedback_collector.py
from typing import Dict, List
from dataclasses import dataclass, field
from datetime import datetime

@dataclass
class UserFeedback:
    question_id: str
    user_id: str
    rating: int  # 1-5
    usefulness: int  # 1-5
    comments: str
    created_at: datetime = field(default_factory=datetime.now)

class FeedbackCollector:
    """L5: 反馈闭环"""

    def collect(self, question_id: str, user_id: str, feedback: Dict) -> UserFeedback:
        """收集用户反馈"""
        return UserFeedback(
            question_id=question_id,
            user_id=user_id,
            rating=feedback.get("rating", 3),
            usefulness=feedback.get("usefulness", 3),
            comments=feedback.get("comments", "")
        )

    def analyze(self, feedback_list: List[UserFeedback]) -> Dict:
        """分析反馈"""
        if not feedback_list:
            return {"status": "no_data"}

        avg_rating = sum(f.rating for f in feedback_list) / len(feedback_list)
        avg_usefulness = sum(f.usefulness for f in feedback_list) / len(feedback_list)

        # 识别问题
        low_rated = [f for f in feedback_list if f.rating <= 2]

        return {
            "total_count": len(feedback_list),
            "avg_rating": avg_rating,
            "avg_usefulness": avg_usefulness,
            "low_rated_count": len(low_rated),
            "issues": [f.comments for f in low_rated if f.comments],
            "status": "needs_improvement" if avg_rating < 3 else "good"
        }
```

```python
from feedback_collector import FeedbackCollector

class GateLevel(Enum):
    L0_INPUT = "L0_INPUT"
    L1_PLAN = "L1_PLAN"
    L2_PROCESS = "L2_PROCESS"
    L3_RESULT = "L3_RESULT"
    L4_DELIVERY = "L4_DELIVERY"
    L5_FEEDBACK = "L5_FEEDBACK"

class QualityGateManager:
    """质量门禁管理器"""

    def __init__(self):
        self.validators = {
            GateLevel.L0_INPUT: InputValidator(),
            GateLevel.L1_PLAN: PlanApprover(),
            GateLevel.L2_PROCESS: ProcessChecker(),
            GateLevel.L3_RESULT: ResultReviewer(),
            GateLevel.L4_DELIVERY: DeliveryApprover(),
            GateLevel.L5_FEEDBACK: FeedbackCollector()
        }

    def pass_through(self, level: GateLevel, data: Dict) -> Dict:
        """通过门禁"""
        validator = self.validators.get(level)

        if not validator:
            return {"status": "error", "message": f"Unknown gate: {level}"}

        # 根据门禁级别调用不同方法
        if level == GateLevel.L0_INPUT:
            result = validator.validate(
                data.get("question", ""),
                data.get("context")
            )
        elif level == GateLevel.L1_PLAN:
            result = validator.approve(
                data.get("intent", {}),
                data.get("experts", []),
                data.get("mode", "parallel")
            )
        elif level == GateLevel.L2_PROCESS:
            result = validator.check(data.get("task_states", {}))
        elif level == GateLevel.L3_RESULT:
            result = validator.review(data.get("outputs", {}))
        elif level == GateLevel.L4_DELIVERY:
            result = validator.approve(
                data.get("final_report", {}),
                data.get("expert_outputs", {})
            )
        elif level == GateLevel.L5_FEEDBACK:
            result = validator.collect(
                data.get("question_id"),
                data.get("user_id"),
                data.get("feedback", {})
            )

        # 转换结果
        return {
            "gate": level.value,
            "passed": result.level.value == "pass" if hasattr(result, "level") else result.status.value in ["approved", "pass"],
            "result": result
        }

    def can_proceed(self, level: GateLevel, data: Dict) -> bool:
        """是否可以进入下一阶段"""
        result = self.pass_through(level, data)

        if level == GateLevel.L0_INPUT:
            return result["result"].level.value in ["pass", "warning"]
        elif level == GateLevel.L1_PLAN:
            return result["result"].status.value == "approved"
        elif level == GateLevel.L2_PROCESS:
            return result["result"].status.value in ["normal", "warning"]
        elif level == GateLevel.L3_RESULT:
            # P2修复: L3 now uses Q70 levels (A/B/C/D/F mapped to PASS/NEEDS_REVISION/FAIL)
            # PASS or NEEDS_REVISION can proceed (needs_revision means needs work before delivery)
            return result["result"].status.value in ["pass", "needs_revision"]
        elif level == GateLevel.L4_DELIVERY:
            return result["result"].status.value == "approved"

        return True
```

## 四、CLI命令

### 4.1 门禁命令

```bash
# L0: 校验输入
thinking-team validate "如何提升DAU？"

thinking-team review --outputs q_001

thinking-team deliver --report q_001
```

## 五、总结

### 5.1 五级门禁

| 级别 | 名称 | 检查内容 | 通过条件 |
|------|------|----------|----------|
| L0 | 输入校验 | 问题格式、长度、敏感词 | 无错误 |
| L1 | 计划审批 | 意图清晰、专家匹配、模式合理 | ≥70分 |
| L2 | 过程检查 | 进度正常、无阻塞 | 正常/警告 |
| L3 | 结果评审 | **六维度 (Q70)**: 完整/专业/实用/逻辑/创新/安全 | ≥60分 (C级) |
| L4 | 交付终审 | 报告质量、共识度、风险 | ≥75分+中低风险 |
| L5 | 反馈闭环 | 用户评分、持续改进 | 分析改进 |

**L3 质量等级说明** (Q70 标准):
- A级 (90+): 优秀，直接通过
- B级 (75-89): 良好，微调后通过
- C级 (60-74): 合格，需要修订
- D级 (45-59): 较差，需要重做
- F级 (<45): 不合格，拒绝

### 5.2 核心组件

| 组件 | 级别 | 功能 |
|------|------|------|
| InputValidator | L0 | 校验问题输入 |
| PlanApprover | L1 | 审批执行计划 |
| ProcessChecker | L2 | 监控执行过程 |
| ResultReviewer | L3 | 评审专家输出 |
| DeliveryApprover | L4 | 终审最终交付 |
| FeedbackCollector | L5 | 收集用户反馈 |
