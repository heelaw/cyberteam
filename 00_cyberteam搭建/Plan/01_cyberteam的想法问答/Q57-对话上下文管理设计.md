# Q57: 思考天团对话上下文管理设计

**问题**: 如何设计思考天团的多轮对话上下文管理机制，支持用户多轮追问？

---

## 一、设计目标与原则

### 1.1 核心目标

| 目标 | 说明 |
|------|------|
| **多轮对话支持** | 支持用户连续追问，系统理解对话连续性 |
| **上下文追溯** | 记录完整对话历史，支持回溯和检索 |
| **资源控制** | 控制上下文长度，避免Token浪费 |
| **状态管理** | 跟踪对话状态，区分追问和新建对话 |

### 1.2 设计原则

| 原则 | 说明 |
|------|------|
| **与Q20保持一致** | 基于workspace/tasks目录结构存储上下文 |
| **与Q53保持一致** | 上下文长度受Token预算约束 |
| **最小化原则** | 只保留必要的上下文信息 |
| **可追溯性** | 每个对话轮次可独立追溯 |

### 1.3 与Q20文件体系的关系

```
Q20 文件体系 ─────────────────► 上下文存储基础
     │
     ├── workspace/tasks/{question_id}/
     │   ├── board.json          ← 任务状态（含对话状态）
     │   ├── metadata.json       ← 元信息
     │   └── context/           ← 新增：对话上下文目录
     │       ├── conversation.json
     │       ├── rounds.json
     │       └── summary.json
     │
     └── input/questions/
         └── {question_id}.json ← 问题（含对话链ID）

本设计 ──────────────────────► 对话上下文增强
     │
     ├── 对话会话管理
     ├── 多轮追问处理
     ├── Token预算控制
     └── 上下文检索
```

## 二、对话上下文内容定义

### 2.1 上下文分解树

```
对话上下文 (Conversation Context)
│
├── 会话元信息 (Session Meta)
│   ├── session_id: 会话唯一标识
│   ├── parent_question_id: 原始问题ID
│   ├── current_round: 当前轮次
│   ├── created_at: 会话开始时间
│   └── last_active: 最后活跃时间
│
├── 对话历史 (Dialogue History)
│   ├── rounds[]
│   │   ├── round_id: 轮次ID
│   │   ├── user_question: 用户问题
│   │   ├── system_response: 系统回答
│   │   ├── expert_outputs: 专家输出
│   │   ├── token_usage: Token消耗
│   │   └── timestamp: 时间戳
│   │
│   └── summary: 对话摘要
│
├── 上下文引用 (Context References)
│   ├── referenced_experts: 引用的专家
│   ├── referenced_findings: 引用的发现
│   └── referenced_decisions: 引用的决策
│
└── 用户画像 (User Profile)
    ├── user_id: 用户ID
    ├── user_preferences: 用户偏好
    └── interaction_patterns: 交互模式
```

### 2.2 上下文内容详情

| 内容类型 | 字段 | 说明 | 是否必须 |
|----------|------|------|----------|
| **会话ID** | session_id | 唯一标识一个多轮对话 | 必须 |
| **原始问题** | original_question | 首次提问内容 | 必须 |
| **轮次记录** | rounds[] | 每轮问答对 | 必须 |
| **对话摘要** | summary | 累计摘要（用于长对话） | 建议 |
| **引用发现** | findings | 之前专家的关键发现 | 可选 |
| **用户偏好** | preferences | 用户交互偏好 | 可选 |

### 3.3 检索机制

#### 3.3.1 会话检索

```python
# context_manager.py
class ContextManager:
    """对话上下文管理器"""

    CONTEXT_DIR = "workspace/tasks/{question_id}/context"

    def get_session(self, session_id: str) -> dict:
        """获取会话"""
        session_file = f"{self.CONTEXT_DIR.format(question_id=self._get_question_id(session_id))}/conversation.json"
        return self._read_json(session_file)

    def get_round(self, session_id: str, round_id: int) -> dict:
        """获取指定轮次"""
        round_file = f"{self.CONTEXT_DIR}/rounds/round_{round_id:03d}.json"
        return self._read_json(round_file)

    def get_latest_rounds(self, session_id: str, count: int = 2) -> list:
        """获取最近N轮对话"""
        session = self.get_session(session_id)
        current_round = session.get("current_round", 0)

        rounds = []
        for i in range(max(1, current_round - count + 1), current_round + 1):
            rounds.append(self.get_round(session_id, i))

        return rounds

    def get_summary(self, session_id: str) -> dict:
        """获取对话摘要"""
        summary_file = f"{self.CONTEXT_DIR}/summary.json"
        return self._read_json(summary_file)

    def search_findings(self, session_id: str, keyword: str) -> list:
        """搜索关键发现"""
        results = []
        session = self.get_session(session_id)

        for round_id in range(1, session["current_round"] + 1):
            round_data = self.get_round(session_id, round_id)
            findings = round_data.get("key_findings", [])

            for finding in findings:
                if keyword in finding:
                    results.append({
                        "round_id": round_id,
                        "finding": finding,
                        "expert": round_data.get("experts", [{}])[0].get("id")
                    })

        return results
```

#### 3.3.2 上下文构建

```python
# context_builder.py
class ContextBuilder:
    """上下文构建器"""

    def build_context_for_round(self, session_id: str, round_id: int) -> dict:
        """为指定轮次构建上下文"""

        # 1. 加载会话信息
        session = self.context_manager.get_session(session_id)

        # 2. 加载历史轮次
        if round_id == 1:
            # 首次提问，无历史
            return {
                "type": "initial",
                "original_question": session["original_question"]
            }
        else:
            # 追问，加载前N轮
            previous_rounds = self.context_manager.get_latest_rounds(
                session_id,
                count=self._get_context_depth(round_id)
            )

            # 3. 构建上下文
            context = {
                "type": "followup",
                "original_question": session["original_question"],
                "previous_rounds": previous_rounds,
                "accumulated_summary": self.context_manager.get_summary(session_id)
            }

            # 4. Token控制（基于Q53）
            context = self._trim_to_token_limit(context)

            return context

    def _get_context_depth(self, round_id: int) -> int:
        """根据轮次决定上下文深度"""
        depth_config = {
            1: 0,   # 第1轮，无历史
            2: 1,   # 第2轮，加载第1轮
            3: 2,   # 第3轮，加载前2轮
            4: 3,   # 第4轮，加载前3轮
            5+: 3   # 5轮及以后，最多加载前3轮
        }
        return depth_config.get(round_id, 3)
```

```python
# followup_detector.py
class FollowupDetector:
    """追问检测器"""

    FOLLOWUP_INDICATORS = {
        # 疑问词
        "如何": "方法询问",
        "怎么": "方法询问",
        "什么": "内容询问",
        "哪些": "列举询问",
        "为什么": "原因询问",
        "多少": "数量询问",
        "预算": "资源询问",
        "具体": "详细询问",

        # 承接词
        "然后": "流程追问",
        "接下来": "流程追问",
        "还有呢": "扩展追问",
        "除此之外": "扩展追问",

        # 指代词
        "这个": "指代追问",
        "它": "指代追问",
        "上述": "指代追问",
        "刚才": "指代追问"
    }

    def detect(self, question: str, session_id: str = None) -> dict:
        """检测是否为追问"""

        result = {
            "is_followup": False,
            "type": "initial",
            "confidence": 0.0,
            "referenced_rounds": [],
            "referenced_findings": []
        }

        # 规则1: 有会话ID，表示在已有会话中
        if session_id:
            result["is_followup"] = True
            result["type"] = "continuation"
            result["confidence"] = 0.8

        # 规则2: 检测追问关键词
        for keyword, followup_type in self.FOLLOWUP_INDICATORS.items():
            if keyword in question:
                result["is_followup"] = True
                result["type"] = followup_type
                result["confidence"] = max(result["confidence"], 0.6)

        # 规则3: 如果有历史会话，分析语义关联
        if session_id and result["is_followup"]:
            # 分析问题与历史对话的关联
            referenced = self._analyze_references(question, session_id)
            result["referenced_rounds"] = referenced["rounds"]
            result["referenced_findings"] = referenced["findings"]

        return result

    def _analyze_references(self, question: str, session_id: str) -> dict:
        """分析问题引用的历史内容"""
        # 查找问题中引用的之前发现
        session = self.context_manager.get_session(session_id)

        referenced_rounds = []
        referenced_findings = []

        for round_id in range(1, session["current_round"] + 1):
            round_data = self.context_manager.get_round(session_id, round_id)
            findings = round_data.get("key_findings", [])

            # 简单匹配：检查问题是否与发现相关
            for finding in findings:
                # 这里可以用更复杂的语义匹配
                if any(word in question for word in finding.split()):
                    referenced_rounds.append(round_id)
                    referenced_findings.append(finding)

        return {
            "rounds": list(set(referenced_rounds)),
            "findings": referenced_findings
        }
```

### 4.2 追问路由

#### 4.2.1 路由决策

```python
# followup_router.py
class FollowupRouter:
    """追问路由"""

    def route(self, question: str, context: dict) -> dict:
        """决定如何处理追问"""

        # 1. 检测追问类型
        detection = self.followup_detector.detect(
            question,
            context.get("session_id")
        )

        # 2. 根据类型决定处理方式
        if detection["type"] == "方法询问":
            # 需要详细执行方案
            return {
                "mode": "detail",
                "experts": ["grow", "first_principle"],
                "context_depth": 2,
                "focus": "execution"
            }

        elif detection["type"] == "资源询问":
            # 需要成本/预算分析
            return {
                "mode": "analysis",
                "experts": ["six_hats", "pmo"],
                "context_depth": 1,
                "focus": "resources"
            }

        elif detection["type"] == "原因询问":
            # 需要深层分析
            return {
                "mode": "analysis",
                "experts": ["kahneman", "first_principle"],
                "context_depth": 2,
                "focus": "reasoning"
            }

        elif detection["type"] == "指代追问":
            # 需要引用之前内容
            return {
                "mode": "reference",
                "experts": [],
                "context_depth": "all",
                "focus": "clarification"
            }

        else:
            # 默认处理
            return {
                "mode": "general",
                "experts": ["grow"],
                "context_depth": 1,
                "focus": "general"
            }
```

## 五、上下文长度控制

### 5.1 Token预算体系（基于Q53）

#### 5.1.1 分层预算

| 层级 | 预算 | 说明 |
|------|------|------|
| **L1 单轮预算** | 50,000 tokens | 首次提问的最大Token |
| **L2 追问预算** | 30,000 tokens | 单次追问的最大Token |
| **L3 会话总预算** | 200,000 tokens | 单个会话的总Token上限 |

#### 5.1.2 预算分配

```python
# token_budget.py
class TokenBudget:
    """Token预算管理器"""

    BUDGET_LIMITS = {
        "initial": 50000,      # 首次提问
        "followup": 30000,     # 追问
        "session_total": 200000  # 会话总计
    }

    def calculate_budget(self, session_id: str, round_type: str) -> dict:
        """计算可用预算"""

        # 1. 获取基础预算
        base_budget = self.BUDGET_LIMITS.get(round_type, 30000)

        # 2. 计算已使用
        used = self._get_session_usage(session_id)

        # 3. 计算剩余
        remaining = self.BUDGET_LIMITS["session_total"] - used

        # 4. 取最小值
        available = min(base_budget, remaining)

        return {
            "base_budget": base_budget,
            "used": used,
            "remaining": remaining,
            "available": available,
            "will_exceed": used + base_budget > self.BUDGET_LIMITS["session_total"]
        }

    def _get_session_usage(self, session_id: str) -> int:
        """获取会话已用Token"""
        session = self.context_manager.get_session(session_id)
        total = 0

        for round_id in range(1, session["current_round"] + 1):
            round_data = self.context_manager.get_round(session_id, round_id)
            total += round_data.get("token_usage", {}).get("total", 0)

        return total
```

### 5.2 上下文压缩策略

#### 5.2.1 压缩触发条件

| 条件 | 动作 |
|------|------|
| 单轮超预算 | 压缩当前轮上下文 |
| 会话超预算 | 压缩历史轮次 + 告警 |
| 超过5轮 | 启用摘要模式 |

#### 5.2.2 压缩方法

```python
# context_compressor.py
class ContextCompressor:
    """上下文压缩器"""

    COMPRESSION_STRATEGIES = {
        "summary": self._compress_to_summary,
        "key_findings": self._compress_to_findings,
        "truncate": self._compress_by_truncation
    }

    def compress(self, context: dict, target_tokens: int) -> dict:
        """压缩上下文到目标Token数"""

        current_tokens = self._estimate_tokens(context)

        if current_tokens <= target_tokens:
            return context  # 无需压缩

        # 策略1: 摘要模式（优先）
        if current_tokens > target_tokens * 1.5:
            return self._compress_to_summary(context, target_tokens)

        # 策略2: 关键发现模式
        elif current_tokens > target_tokens * 1.2:
            return self._compress_to_findings(context, target_tokens)

        # 策略3: 截断模式（最后手段）
        else:
            return self._compress_by_truncation(context, target_tokens)

    def _compress_to_summary(self, context: dict, target_tokens: int) -> dict:
        """压缩为摘要"""

        # 保留: 原始问题 + 摘要 + 当前轮次
        return {
            "type": "summary_mode",
            "original_question": context.get("original_question"),
            "accumulated_summary": context.get("accumulated_summary"),
            "current_round": context.get("current_round"),
            "note": "历史轮次已压缩为摘要"
        }

    def _compress_to_findings(self, context: dict, target_tokens: int) -> dict:
        """压缩为关键发现"""

        compressed = {
            "type": "findings_mode",
            "original_question": context.get("original_question"),
            "key_findings": [],
            "decisions": []
        }

        # 提取所有关键发现
        for round_data in context.get("previous_rounds", []):
            findings = round_data.get("key_findings", [])
            compressed["key_findings"].extend(findings)

            decisions = round_data.get("decisions", [])
            compressed["decisions"].extend(decisions)

        # 限制数量
        compressed["key_findings"] = compressed["key_findings"][:10]
        compressed["decisions"] = compressed["decisions"][:5]

        return compressed

    def _compress_by_truncation(self, context: dict, target_tokens: int) -> dict:
        """通过截断压缩"""

        # 只保留最近2轮
        if "previous_rounds" in context:
            context["previous_rounds"] = context["previous_rounds"][-2:]

        # 截断专家输出
        for round_data in context.get("previous_rounds", []):
            for expert in round_data.get("experts", []):
                if "output" in expert:
                    # 保留前1000字符
                    expert["output"] = expert["output"][:1000] + "..."

        return context

    def _estimate_tokens(self, context: dict) -> int:
        """估算Token数（约等于字符数/4）"""
        import json
        content = json.dumps(context, ensure_ascii=False)
        return len(content) // 4
```
