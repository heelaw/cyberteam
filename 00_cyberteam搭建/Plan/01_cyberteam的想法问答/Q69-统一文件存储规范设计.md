```yaml
# 命名规则

  format: "task_{expert_id}_{question_id}"
```

## 四、访问接口

### 4.1 文件访问器

```python
# file_accessor.py
from pathlib import Path
from typing import Dict, List, Optional
import json

class FileAccessor:
    """统一文件访问器"""

    ROOT = Path("thinking-team")

    # 基础路径映射
    BASE_PATHS = {
        "input_questions": ROOT / "input" / "questions",
        "input_context": ROOT / "input" / "context",
        "output_reports": ROOT / "output" / "reports",
        "output_decisions": ROOT / "output" / "decisions",
        "workspace_tasks": ROOT / "workspace" / "tasks",
        "workspace_messages": ROOT / "workspace" / "messages",
        "knowledge_patterns": ROOT / "knowledge" / "patterns",
        "knowledge_experts": ROOT / "knowledge" / "experts",
        "logs_execution": ROOT / "logs" / "execution",
    }

    @classmethod
    def get_question(cls, question_id: str) -> Dict:
        """获取问题"""
        path = cls.BASE_PATHS["input_questions"] / "pending" / f"{question_id}.json"
        return cls._read_json(path)

    @classmethod
    def save_question(cls, question_id: str, data: Dict, status: str = "pending"):
        """保存问题"""
        path = cls.BASE_PATHS["input_questions"] / status / f"{question_id}.json"
        cls._write_json(path, data)

    @classmethod
    def get_board(cls, question_id: str) -> Dict:
        """获取看板状态"""
        path = cls.BASE_PATHS["workspace_tasks"] / question_id / "board.json"
        return cls._read_json(path)

    @classmethod
    def save_board(cls, question_id: str, data: Dict):
        """保存看板状态"""
        path = cls.BASE_PATHS["workspace_tasks"] / question_id / "board.json"
        cls._ensure_dir(path.parent)
        cls._write_json(path, data)

    @classmethod
    def get_expert_output(cls, question_id: str, expert_id: str) -> str:
        """获取专家输出"""
        path = cls.BASE_PATHS["workspace_tasks"] / question_id / "tasks" / expert_id / "output.md"
        return path.read_text()

    @classmethod
    def get_report(cls, question_id: str) -> Dict:
        """获取报告"""
        path = cls.BASE_PATHS["output_reports"] / question_id / "报告.md"
        return {"content": path.read_text()}

    @classmethod
    def _read_json(cls, path: Path) -> Dict:
        """读取JSON"""
        if not path.exists():
            return {}
        return json.loads(path.read_text())

    @classmethod
    def _write_json(cls, path: Path, data: Dict):
        """写入JSON"""
        cls._ensure_dir(path.parent)
        path.write_text(json.dumps(data, ensure_ascii=False, indent=2))

    @classmethod
    def _ensure_dir(cls, path: Path):
        """确保目录存在"""
        path.mkdir(parents=True, exist_ok=True)
```

## 五、路径冲突解决

### 5.1 冲突对照表

| 原冲突 | 解决 | 统一路径 |
|--------|------|----------|
| Q20: `workspace/tasks/` | 统一 | `thinking-team/workspace/tasks/{question_id}/` |
| Q41: `logs/tasks/` | 合并到 | `thinking-team/workspace/tasks/{question_id}/tasks/` |
| Q57: `workspace/tasks/{question_id}/context/` | 合并到 | `thinking-team/workspace/tasks/{question_id}/context/` |
| Q20: `output/reports/` | 统一 | `thinking-team/output/reports/{question_id}/` |
| Q41: `logs/execution/` | 统一 | `thinking-team/logs/execution/{question_id}/` |

### 5.2 迁移脚本

```python
# migrate_storage.py
from pathlib import Path
import shutil
import json

def migrate_old_to_new():
    """旧路径迁移到新路径"""

    mappings = [
        # Q20 → Q69
        ("old/workspace/tasks/", "thinking-team/workspace/tasks/"),
        ("old/output/reports/", "thinking-team/output/reports/"),
        ("old/logs/execution/", "thinking-team/logs/execution/"),

        # Q41 → Q69
        ("old/logs/tasks/", "thinking-team/workspace/tasks/"),

        # Q57 → Q69 (Q57实际使用 workspace/tasks/{id}/context/)
        ("old/workspace/tasks/", "thinking-team/workspace/tasks/"),
    ]

    for old_prefix, new_prefix in mappings:
        migrate_directory(old_prefix, new_prefix)

def migrate_directory(old: str, new: str):
    """迁移目录"""
    old_path = Path(old)
    new_path = Path(new)

    if not old_path.exists():
        return

    # 迁移文件
    for file in old_path.rglob("*"):
        if file.is_file():
            relative = file.relative_to(old_path)
            new_file = new_path / relative

            # 创建目录
            new_file.parent.mkdir(parents=True, exist_ok=True)

            # 移动文件
            shutil.move(str(file), str(new_file))

    print(f"Migrated: {old} → {new}")
```

## 六、总结

### 6.1 统一规范

| 维度 | 规范 |
|------|------|
| **根目录** | `thinking-team/` |
| **ID格式** | `q_YYYYMMDD_HHMMSS_xxx` |
| **状态后缀** | `pending/processing/completed` |
| **JSON Schema** | 所有状态文件使用Schema |

### 6.2 核心原则

1. **单一来源**: 每个数据只有唯一存储位置
2. **层级清晰**: input → workspace → output → knowledge
3. **命名一致**: 统一的ID格式和文件命名
4. **Schema验证**: 所有JSON文件遵循Schema
