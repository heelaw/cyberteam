### 1.2 各阶段评分权重

| 阶段 | 权重调整 | 说明 |
|------|----------|------|
| **L3: 结果评审** | 完整性×1.0 + 专业性×1.0 + 实用性×1.0 + 逻辑性×0.75 + 创新性×0.5 + 安全性×0.3 | 重点在完整、专业、实用 |
| **L4: 交付终审** | 完整性×1.0 + 实用性×1.5 + 逻辑性×1.0 + 安全性×1.0 | 重点在实用和安全 |
| **L5: 反馈评估** | 完整性×0.5 + 实用性×1.5 + 用户满意度×1.5 + 创新性×0.5 | 重点在用户价值 |

## 二、评分细则

### 2.1 完整性 (25分)

```python
class CompletenessScorer:
    """完整性评分"""

    def score(self, output: Dict) -> float:
        """
        评分标准:
        - 核心观点: 10分
        - 分析过程: 5分
        - 建议/方案: 5分
        - 风险/注意事项: 5分
        """

        content = output.get("output", "")
        score = 0

        # 核心观点 (10分)
        if any(kw in content for kw in ["核心", "观点", "结论", "主要"]):
            score += 10

        # 分析过程 (5分)
        if "##" in content or "1." in content:  # 有结构
            score += 5
        elif len(content) > 500:
            score += 3

        # 建议/方案 (5分)
        if any(kw in content for kw in ["建议", "方案", "可以", "应该"]):
            score += 5

        # 风险/注意事项 (5分)
        if any(kw in content for kw in ["风险", "注意", "问题", "挑战"]):
            score += 5

        return min(score, 25)
```

### 2.2 专业性 (25分)

```python
class ProfessionalismScorer:
    """专业性评分"""

    # 专家框架关键词
    FRAMEWORK_KEYWORDS = {
        "kahneman": ["认知偏差", "锚定效应", "损失厌恶", "确认偏差", "心理账户"],
        "first_principle": ["本质", "从零", "重构", "假设", "基本要素"],
        "six_hats": ["白帽", "黑帽", "红帽", "绿帽", "黄帽", "蓝帽"],
        "grow": ["目标", "现状", "选择", "路径", "意愿"],
        "reverse_thinking": ["风险", "失败", "问题", "漏洞", "避免"],
        "swot_tows": ["优势", "劣势", "机会", "威胁"],
        "mckinsey": ["MECE", "金字塔", "假设驱动", "结构化"],
        "five_why": ["为什么", "根本原因", "因果链"],
        "wbs": ["分解", "任务", "里程碑", "交付物"]
    }

    def score(self, output: Dict) -> float:
        """评分标准:
        - 框架关键词: 15分
        - 框架应用准确: 10分
        """

        content = output.get("output", "")
        expert_id = output.get("expert_id", "")

        score = 0

        # 检查框架关键词
        keywords = self.FRAMEWORK_KEYWORDS.get(expert_id, [])
        matched = sum(1 for kw in keywords if kw in content)

        if matched >= 3:
            score += 15
        elif matched >= 1:
            score += 10
        else:
            # 通用专业词
            if any(kw in content for kw in ["分析", "评估", "建议", "方案"]):
                score += 5

        # 检查框架应用
        if matched >= 2:
            score += 10
        elif matched >= 1:
            score += 5

        return min(score, 25)
```

### 2.3 实用性 (20分)

```python
class PracticalityScorer:
    """实用性评分"""

    def score(self, output: Dict) -> float:
        """评分标准:
        - 可执行建议: 10分
        - 具体行动项: 5分
        - 时间/责任人: 5分
        """

        content = output.get("output", "")
        score = 0

        # 可执行建议 (10分)
        if any(kw in content for kw in ["建议", "方案", "可以", "应该"]):
            score += 10

            # 检查具体性
            if len(content) > 300:
                score += 3
            if any(kw in content for kw in ["第一步", "具体", "明确"]):
                score += 2

        # 具体行动项 (5分)
        if any(kw in content for kw in ["步骤", "流程", "方法"]):
            score += 5

        # 时间/责任人 (5分)
        if any(kw in content for kw in ["责任", "负责", "时间", "期限", "周", "月"]):
            score += 5

        return min(score, 20)
```

### 2.4 逻辑性 (15分)

```python
class LogicScorer:
    """逻辑性评分"""

    def score(self, output: Dict) -> float:
        """评分标准:
        - 结构清晰: 5分
        - 因果关系: 5分
        - 结论明确: 5分
        """

        content = output.get("output", "")
        score = 0

        # 结构清晰 (5分)
        if content.count("\n") > 5:
            score += 3
        if "##" in content:
            score += 2

        # 因果关系 (5分)
        if any(kw in content for kw in ["因为", "所以", "因此", "导致", "由于"]):
            score += 5

        # 结论明确 (5分)
        if any(kw in content for kw in ["结论", "总结", "总之", "所以"]):
            score += 5

        return min(score, 15)
```

### 2.5 创新性 (10分)

```python
class InnovationScorer:
    """创新性评分"""

    def score(self, output: Dict) -> float:
        """评分标准:
        - 独特视角: 5分
        - 新见解: 5分
        """

        content = output.get("output", "")
        score = 0

        # 独特视角
        if any(kw in content for kw in ["独特", "创新", "突破", "不同"]):
            score += 5

        # 新见解 (基于长度和内容)
        if len(content) > 800 and "但是" in content:
            # 有转折说明有不同角度
            score += 3
        if "可能" in content or "也许" in content:
            # 有探索性表述
            score += 2

        return min(score, 10)
```

class SafetyScorer:

            score += 3

        # 注意事项
        if any(kw in content for kw in ["但是", "然而", "需注意", "前提"]):
            score += 2

        return min(score, 5)

## 三、统一评分器

### 3.1 UnifiedQualityScorer

```python
# unified_quality_scorer.py
from typing import Dict, Optional, List
from dataclasses import dataclass
from enum import Enum

@dataclass
class QualityScore:
    """统一质量评分"""
    total: float  # 0-100
    breakdown: Dict[str, float]  # 各维度分数
    level: str  # A/B/C/D/F
    issues: List[str]  # 问题列表

class QualityLevel(Enum):
    A = "A"  # 90-100
    B = "B"  # 75-89
    C = "C"  # 60-74
    D = "D"  # 45-59
    F = "F"  # <45

class UnifiedQualityScorer:
    """统一质量评分器"""

    # 分数阈值
    THRESHOLDS = {
        "A": 90,
        "B": 75,
        "C": 60,
        "D": 45
    }

    def __init__(self):
        self.scorers = {
            "completeness": CompletenessScorer(),
            "professionalism": ProfessionalismScorer(),
            "practicality": PracticalityScorer(),
            "logic": LogicScorer(),
            "innovation": InnovationScorer(),
            "safety": SafetyScorer()
        }

    def score(
        self,
        output: Dict,
        stage: str = "result_review"
    ) -> QualityScore:
        """
        评分

        Args:
            output: 专家输出
            stage: 评审阶段 (result_review/delivery/feedback)
        """

        # 各维度评分
        breakdown = {}
        issues = []

        for dimension, scorer in self.scorers.items():
            dimension_score = scorer.score(output)
            breakdown[dimension] = dimension_score

            # 记录问题
            if dimension_score < self._get_threshold(dimension):
                issues.append(f"{dimension}不足 ({dimension_score})")

        # 计算总分
        total = sum(breakdown.values())

        # 根据阶段调整权重
        total = self._adjust_by_stage(total, breakdown, stage)

        # 确定等级
        level = self._get_level(total)

        return QualityScore(
            total=total,
            breakdown=breakdown,
            level=level.value,
            issues=issues
        )

    def _get_threshold(self, dimension: str) -> float:
        """获取维度阈值"""
        thresholds = {
            "completeness": 18,  # 25*0.7
            "professionalism": 18,
            "practicality": 14,  # 20*0.7
            "logic": 10,         # 15*0.7
            "innovation": 7,      # 10*0.7
            "safety": 3           # 5*0.7
        }
        return thresholds.get(dimension, 10)

    def _adjust_by_stage(self, total: float, breakdown: Dict, stage: str) -> float:
        """根据阶段调整分数"""

        if stage == "result_review":
            # 重点: 完整、专业、实用
            adjusted = (
                breakdown["completeness"] * 1.0 +
                breakdown["professionalism"] * 1.0 +
                breakdown["practicality"] * 1.0 +
                breakdown["logic"] * 0.75 +
                breakdown["innovation"] * 0.5 +
                breakdown["safety"] * 0.3
            )
            # 归一化到100
            # P1修正: 实际权重和 = 1.0+1.0+1.0+0.75+0.5+0.3 = 3.55，原注释/代码错误写为3.75
            return adjusted / 3.55 * 100

        elif stage == "delivery":
            # 重点: 实用、安全
            adjusted = (
                breakdown["completeness"] * 1.0 +
                breakdown["professionalism"] * 0.5 +
                breakdown["practicality"] * 1.5 +
                breakdown["logic"] * 1.0 +
                breakdown["innovation"] * 0.3 +
                breakdown["safety"] * 1.0
            )
            return adjusted / 4.3 * 100

        else:
            return total

    def _get_level(self, total: float) -> QualityLevel:
        """获取等级"""
        if total >= self.THRESHOLDS["A"]:
            return QualityLevel.A
        elif total >= self.THRESHOLDS["B"]:
            return QualityLevel.B
        elif total >= self.THRESHOLDS["C"]:
            return QualityLevel.C
        elif total >= self.THRESHOLDS["D"]:
            return QualityLevel.D
        else:
            return QualityLevel.F
```

## 六、总结

### 6.1 统一规范

| 维度 | 满分 | 阈值 | 说明 |
|------|------|------|------|
| 完整性 | 25 | 18 | 核心观点+分析+建议+风险 |
| 专业性 | 25 | 18 | 框架关键词+准确应用 |
| 实用性 | 20 | 14 | 可执行+具体+可落地 |
| 逻辑性 | 15 | 10 | 结构+因果+结论 |
| 创新性 | 10 | 7 | 独特视角+新见解 |
| 安全性 | 5 | 3 | 风险提示+注意事项 |
| **总分** | **100** | **70** | A≥90, B≥75, C≥60 |

### 6.2 等级定义

| 等级 | 分数 | 含义 | 行动 |
|------|------|------|------|
| A | 90+ | 优秀 | 直接通过 |
| B | 75-89 | 良好 | 微调后通过 |
| C | 60-74 | 合格 | 需要修订 |
| D | 45-59 | 较差 | 需要重做 |
| F | <45 | 不合格 | 拒绝 |
