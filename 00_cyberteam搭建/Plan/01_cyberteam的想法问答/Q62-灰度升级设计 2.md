## 一、灰度升级体系设计原则

### 1.1 设计目标

| 目标 | 说明 |
|------|------|
| **风险可控** | 新专家上线问题仅影响小范围用户 |
| **平滑过渡** | 用户无感知切换，不影响体验连续性 |
| **快速验证** | 缩短新专家的验证周期 |
| **快速回滚** | 问题发生时能够快速恢复到稳定版本 |
| **数据驱动** | 基于实际效果数据做升级决策 |

```python
# grayscale/state_machine.py

from enum import Enum
from dataclasses import dataclass
from datetime import datetime
from typing import Optional, List

class GrayscaleStage(Enum):
    """灰度阶段"""
    INTERNAL_TEST = "internal_test"      # 内部测试 (0%)
    SMALL_PILOT = "small_pilot"          # 小流量 (5%)
    MEDIUM_PILOT = "medium_pilot"        # 中流量 (30%)
    LARGE_PILOT = "large_pilot"          # 大流量 (70%)
    FULL_RELEASE = "full_release"        # 全量发布 (100%)

class GrayscaleStatus(Enum):
    """灰度状态"""
    PENDING = "pending"           # 待激活
    RUNNING = "running"          # 运行中
    PAUSED = "paused"            # 暂停
    ROLLBACK = "rollback"        # 回滚中
    COMPLETED = "completed"       # 已完成

@dataclass
class GrayscaleConfig:
    """灰度配置"""
    expert_id: str               # 专家ID
    version: str                # 版本号
    stage: GrayscaleStage       # 当前阶段
    status: GrayscaleStatus      # 当前状态
    start_time: datetime        # 开始时间
    target_percentage: int      # 目标比例
    current_percentage: int     # 当前比例

    # 各阶段配置
    stage_config = {
        GrayscaleStage.INTERNAL_TEST: {
            "percentage": 0,
            "duration_days": 5,
            "conditions": ["功能测试通过", "性能测试通过"]
        },
        GrayscaleStage.SMALL_PILOT: {
            "percentage": 5,
            "duration_days": 3,
            "conditions": ["错误率<1%", "响应时间<1.5x基线", "无P0/P1告警"]
        },
        GrayscaleStage.MEDIUM_PILOT: {
            "percentage": 30,
            "duration_days": 5,
            "conditions": ["错误率<0.5%", "响应时间<1.2x基线", "满意度>85%"]
        },
        GrayscaleStage.LARGE_PILOT: {
            "percentage": 70,
            "duration_days": 7,
            "conditions": ["错误率<0.3%", "响应时间<1.1x基线", "满意度>90%"]
        },
        GrayscaleStage.FULL_RELEASE: {
            "percentage": 100,
            "duration_days": 0,  # 持续运行
            "conditions": []
        }
    }

class GrayscaleStateMachine:
    """灰度状态机"""

    # 阶段转换映射
    TRANSITIONS = {
        GrayscaleStage.INTERNAL_TEST: GrayscaleStage.SMALL_PILOT,
        GrayscaleStage.SMALL_PILOT: GrayscaleStage.MEDIUM_PILOT,
        GrayscaleStage.MEDIUM_PILOT: GrayscaleStage.LARGE_PILOT,
        GrayscaleStage.LARGE_PILOT: GrayscaleStage.FULL_RELEASE,
    }

    def __init__(self, config: GrayscaleConfig):
        self.config = config
        self.evaluation_history: List[dict] = []

    def can_promote(self, metrics: dict) -> bool:
        """检查是否可以升级到下一阶段"""
        current_stage = self.config.stage
        conditions = self.config.stage_config[current_stage]["conditions"]

        # 检查所有条件是否满足
        for condition in conditions:
            if not self._check_condition(condition, metrics):
                return False

        # 检查最小持续时间
        duration = self.config.stage_config[current_stage]["duration_days"]
        if duration > 0:
            elapsed = (datetime.now() - self.config.start_time).days
            if elapsed < duration:
                return False

        return True

    def _check_condition(self, condition: str, metrics: dict) -> bool:
        """检查单个条件"""
        # 解析条件并检查指标
        if "错误率" in condition:
            threshold = float(condition.split("<")[1].replace("%", ""))
            return metrics.get("error_rate", 100) < threshold

        if "响应时间" in condition:
            threshold = float(condition.split("<")[1].split("x")[0])
            baseline = metrics.get("response_time_baseline", 1)
            current = metrics.get("response_time", baseline * 10)
            return current < baseline * threshold

        if "满意度" in condition:
            threshold = float(condition.split(">")[1].replace("%", ""))
            return metrics.get("user_satisfaction", 0) >= threshold

        if "P0" in condition:
            return metrics.get("p0_alerts", 0) == 0

        if "P1" in condition:
            return metrics.get("p1_alerts", 999) == 0

        return True
```

        if self._is_internal_user(user_id):

        if self._is_core_user(user_id):

        # 普通用户按比例启用
        return bucket < percentage

# grayscale/monitor.py

from dataclasses import dataclass
from typing import List, Dict, Optional
from datetime import datetime, timedelta

@dataclass
class GrayscaleMetrics:
    """灰度指标"""
    expert_id: str
    version: str
    stage: str

    # 稳定性指标
    total_requests: int = 0
    error_count: int = 0
    timeout_count: int = 0
    crash_count: int = 0

    # 性能指标
    response_time_p50: float = 0
    response_time_p99: float = 0
    throughput: float = 0

    # 质量指标
    quality_pass_rate: float = 0
    user_satisfaction: float = 0

    # 业务指标
    conversion_rate: float = 0
    retention_rate: float = 0

    # 时间信息
    timestamp: datetime = None

    @property
    def error_rate(self) -> float:
        if self.total_requests == 0:
            return 0
        return self.error_count / self.total_requests

    @property
    def timeout_rate(self) -> float:
        if self.total_requests == 0:
            return 0
        return self.timeout_count / self.total_requests


class GrayscaleMonitor:
    """灰度监控"""

    def __init__(self, expert_id: str):
        self.expert_id = expert_id
        self.metrics_history: List[GrayscaleMetrics] = []

    def collect_metrics(self) -> GrayscaleMetrics:
        """收集当前指标"""
        # 从各个数据源汇总指标
        metrics = GrayscaleMetrics(
            expert_id=self.expert_id,
            version=self._get_current_version(),
            stage=self._get_current_stage(),
            timestamp=datetime.now()
        )

        # 1. 从任务日志获取稳定性指标
        metrics.total_requests = self._get_total_requests()
        metrics.error_count = self._get_error_count()
        metrics.timeout_count = self._get_timeout_count()

        # 2. 从性能监控获取性能指标
        metrics.response_time_p50 = self._get_p50_latency()
        metrics.response_time_p99 = self._get_p99_latency()
        metrics.throughput = self._get_throughput()

        # 3. 从质量评估获取质量指标
        metrics.quality_pass_rate = self._get_quality_pass_rate()
        metrics.user_satisfaction = self._get_user_satisfaction()

        # 4. 从业务系统获取业务指标
        metrics.conversion_rate = self._get_conversion_rate()
        metrics.retention_rate = self._get_retention_rate()

        self.metrics_history.append(metrics)
        return metrics

    def check_alerts(self, metrics: GrayscaleMetrics) -> List[dict]:
        """检查告警"""
        alerts = []

        # P0 告警
        if metrics.error_rate > 0.05:
            alerts.append({
                "level": "P0",
                "type": "ERROR_RATE_HIGH",
                "message": f"错误率过高: {metrics.error_rate:.2%}",
                "threshold": "5%",
                "current": f"{metrics.error_rate:.2%}"
            })

        if metrics.crash_count > 0:
            alerts.append({
                "level": "P0",
                "type": "CRASH_DETECTED",
                "message": f"检测到崩溃: {metrics.crash_count}次",
                "threshold": "0次",
                "current": f"{metrics.crash_count}次"
            })

        # P1 告警
        if metrics.error_rate > 0.01:
            alerts.append({
                "level": "P1",
                "type": "ERROR_RATE_WARNING",
                "message": f"错误率告警: {metrics.error_rate:.2%}",
                "threshold": "1%",
                "current": f"{metrics.error_rate:.2%}"
            })

        if metrics.user_satisfaction < 0.7:
            alerts.append({
                "level": "P1",
                "type": "SATISFACTION_LOW",
                "message": f"用户满意度低: {metrics.user_satisfaction:.1%}",
                "threshold": "70%",
                "current": f"{metrics.user_satisfaction:.1%}"
            })

        return alerts

### 5.1 决策规则引擎

```python
# grayscale/decision.py

from enum import Enum
from dataclasses import dataclass
from typing import List, Dict, Optional
from datetime import datetime

class DecisionType(Enum):
    """决策类型"""
    PROMOTE = "promote"           # 升级
    PAUSE = "pause"               # 暂停
    ROLLBACK = "rollback"         # 回滚
    CONTINUE = "continue"         # 继续当前阶段

@dataclass
class DecisionRule:
    """决策规则"""
    name: str
    condition: str
    action: DecisionType
    priority: int  # 优先级，数字越小越高

class DecisionEngine:
    """决策引擎"""

    def __init__(self):
        self.rules = self._init_rules()

    def _init_rules(self) -> List[DecisionRule]:
        """初始化决策规则"""
        return [
            # P0 级别 - 立即回滚
            DecisionRule(
                name="P0错误率过高",
                condition="error_rate > 0.05",
                action=DecisionType.ROLLBACK,
                priority=1
            ),
            DecisionRule(
                name="P0崩溃检测",
                condition="crash_count > 0",
                action=DecisionType.ROLLBACK,
                priority=1
            ),
            DecisionRule(
                name="P0服务不可用",
                condition="availability < 0.95",
                action=DecisionType.ROLLBACK,
                priority=1
            ),

            # P1 级别 - 暂停并评估
            DecisionRule(
                name="P1错误率告警",
                condition="error_rate > 0.01",
                action=DecisionType.PAUSE,
                priority=10
            ),
            DecisionRule(
                name="P1满意度低",
                condition="user_satisfaction < 0.7",
                action=DecisionType.PAUSE,
                priority=10
            ),
            DecisionRule(
                name="P1性能下降",
                condition="response_time_p99 > baseline * 3",
                action=DecisionType.PAUSE,
                priority=10
            ),

            # 升级条件
            DecisionRule(
                name="升级条件满足",
                condition="can_promote == true",
                action=DecisionType.PROMOTE,
                priority=100
            ),

            # 默认继续
            DecisionRule(
                name="默认继续",
                condition="true",
                action=DecisionType.CONTINUE,
                priority=999
            )
        ]

    def make_decision(self, metrics: dict, context: dict) -> DecisionType:
        """做出决策"""
        # 按优先级排序规则
        sorted_rules = sorted(self.rules, key=lambda r: r.priority)

        # 依次检查规则
        for rule in sorted_rules:
            if self._evaluate_condition(rule.condition, metrics, context):
                return rule.action

        return DecisionType.CONTINUE

    def _evaluate_condition(self, condition: str, metrics: dict, context: dict) -> bool:
        """评估条件"""
        try:
            # 构建评估上下文
            ctx = {**metrics, **context}

            # 安全评估条件
            return eval(condition, {"__builtins__": {}}, ctx)
        except Exception as e:
            print(f"Condition evaluation error: {e}")
            return False
```

from typing import Optional
from dataclasses import dataclass
from datetime import datetime

@dataclass
class RollbackConfig:
    """回滚配置"""
    expert_id: str
    current_version: str
    target_version: str  # 回滚到上一稳定版本
    reason: str
    initiated_by: str  # system 或 user
    initiated_at: datetime

class RollbackExecutor:
    """回滚执行器"""

    def __init__(self, expert_id: str):
        self.expert_id = expert_id
        self.version_history = self._load_version_history()

    def execute_rollback(self, target_version: Optional[str] = None) -> bool:
        """执行回滚"""
        # 确定回滚目标版本
        if target_version is None:
            target_version = self._get_stable_version()

        # 1. 记录当前状态
        self._save_snapshot()

        # 2. 切换路由
        self._switch_routing(target_version)

        # 3. 通知用户（可选）
        self._notify_users()

        # 4. 更新监控配置
        self._update_monitor_config(target_version)

        # 5. 记录回滚日志
        self._log_rollback(target_version)

        return True

    def _get_stable_version(self) -> str:
        """获取稳定版本"""
        # 从版本历史中查找最近的稳定版本
        for version in reversed(self.version_history):

## 九、总结

### 9.1 设计要点

| 要点 | 说明 |
|------|------|
| **阶段性** | 5个阶段，比例递增：0%→5%→30%→70%→100% |
| **可观测** | 实时监控指标，多维度告警 |
| **可回滚** | 自动检测异常，一键回滚到稳定版本 |
| **数据驱动** | 基于实际效果数据做升级决策 |
| **与现有机制集成** | 复用Q42异常检测、Q49质量评估、Q50反馈闭环 |

### 9.2 关键成功因素

1. **监控全面**: 覆盖稳定性、性能、质量、业务4个维度
2. **决策自动化**: 规则引擎自动判断，减少人工干预
3. **回滚快速**: 问题发生时分钟级回滚
4. **用户稳定**: 一致性哈希保证用户体验一致
