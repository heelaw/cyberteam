# Q65: 并行执行引擎设计

**问题**: 如何设计支持多专家Agent并行执行的引擎？包括任务调度、超时控制、结果收集？

### 2.1 TaskStore（任务存储）— 统一使用 Q64 定义

```python
# Q65 使用 Q64 统一定义的 TaskStore 和 ExpertTask，确保状态模型兼容
#
# 导入来源: Q64-ExpertAgent协议设计.md → expert_agent_protocol/task.py
#
# 关键统一点:
# 1. TaskStatus 枚举与 Q64 完全一致
# 2. ExpertTask 模型是 Q64/Q65 共用的任务格式
# 3. TaskStore 实现 JSON 文件持久化，支持跨组件状态共享

# === 统一类型定义（与 Q64 expert_agent_protocol/task.py 一致）===

import json
import logging
from pathlib import Path
from pydantic import BaseModel, Field
from enum import Enum
from datetime import datetime
from typing import Optional, List, Dict, Any
import time
import uuid

logger = logging.getLogger(__name__)


class TaskStatus(str, Enum):
    """任务状态枚举 - 与 Q64 统一

    状态流转:
    CREATED → QUEUED → RUNNING → COMPLETED
                    ↘ FAILED
                    ↘ CANCELLED
           ↘ WAITING_DEPS（等待依赖）
           ↘ PAUSED（已暂停）
    """
    CREATED = "created"           # 已创建
    QUEUED = "queued"             # 已入队
    RUNNING = "running"           # 执行中
    WAITING_DEPS = "waiting_deps" # 等待依赖
    PAUSED = "paused"             # 已暂停
    COMPLETED = "completed"       # 已完成
    FAILED = "failed"             # 失败
    CANCELLED = "cancelled"      # 已取消


class ExpertTask(BaseModel):
    """ExpertAgent任务模型 - 与 Q64 统一

    Q65 扩展字段:
    - expert_id: 当前执行的专家ID
    - prompt: 专家执行的prompt
    """
    # 核心标识
    task_id: str = Field(..., description="全局唯一任务ID")
    question_id: str = Field(..., description="关联的问题ID")

    # 任务定义
    title: str = ""
    description: str = ""
    task_type: str = "analysis"

    # 状态管理
    status: TaskStatus = TaskStatus.CREATED
    progress: int = Field(0, ge=0, le=100)

    # 执行信息
    created_at: datetime = Field(default_factory=datetime.now)
    started_at: Optional[datetime] = None
    completed_at: Optional[datetime] = None

    # 专家分配
    assigned_experts: List[str] = []
    running_experts: List[str] = []
    completed_experts: List[str] = []

    # 依赖管理
    depends_on: List[str] = []
    blocking: List[str] = []

    # 输出
    output: Dict[str, Any] = Field(default_factory=dict)
    errors: List[str] = []

    # 资源限制
    max_duration_seconds: int = 3600
    max_retries: int = 3
    retry_count: int = 0

    # Q65 扩展字段（Q64 兼容）
    expert_id: str = ""  # 当前执行的专家ID
    prompt: str = ""     # 专家执行的prompt


class TaskStore:
    """任务存储接口 - 基于JSON文件的任务持久化

    统一使用 Q64 的 TaskStore 实现，确保:
    1. Q64/Q65 状态模型完全兼容
    2. 任务状态持久化到 JSON 文件
    3. 支持跨组件状态共享
    """

    def __init__(self, storage_path: str):
        self.storage_path = Path(storage_path)
        self._ensure_dir(self.storage_path)

    def _ensure_dir(self, path: Path):
        """确保目录存在"""
        path.mkdir(parents=True, exist_ok=True)

    def _task_file(self, task_id: str) -> Path:
        """获取任务文件路径"""
        return self.storage_path / f"{task_id}.json"

    def create_task(self, task: ExpertTask) -> ExpertTask:
        """创建任务"""
        task_file = self._task_file(task.task_id)
        with open(task_file, 'w', encoding='utf-8') as f:
            json.dump(task.model_dump(), f, ensure_ascii=False, indent=2)
        return task

    def get_task(self, task_id: str) -> Optional[ExpertTask]:
        """获取任务"""
        task_file = self._task_file(task_id)
        if not task_file.exists():
            return None
        with open(task_file, 'r', encoding='utf-8') as f:
            data = json.load(f)
        return ExpertTask(**data)

    def update_task(self, task: ExpertTask) -> ExpertTask:
        """更新任务"""
        return self.create_task(task)

    def list_tasks(self, status: Optional[TaskStatus] = None) -> List[ExpertTask]:
        """列出任务"""
        tasks = []
        for task_file in self.storage_path.glob("*.json"):
            with open(task_file, 'r', encoding='utf-8') as f:
                data = json.load(f)
            task = ExpertTask(**data)
            if status is None or task.status == status:
                tasks.append(task)
        return tasks

    def delete_task(self, task_id: str) -> bool:
        """删除任务"""
        task_file = self._task_file(task_id)
        if task_file.exists():
            task_file.unlink()
            return True
        return False

    def get_task_dependencies(self, task_id: str) -> List[ExpertTask]:
        """获取任务依赖链"""
        task = self.get_task(task_id)
        if not task:
            return []
        dependencies = []
        for dep_id in task.depends_on:
            dep_task = self.get_task(dep_id)
            if dep_task:
                dependencies.append(dep_task)
        return dependencies

    # === Q65 适配方法 ===

    def get_pending_tasks(self, expert_id: str = None) -> List[ExpertTask]:
        """获取待执行任务 - Q65 适配方法

        替代原来 TaskQueue.get_pending()
        """
        pending = self.list_tasks(status=TaskStatus.QUEUED)
        if expert_id:
            pending = [t for t in pending if expert_id in t.assigned_experts]
        return pending

    def update_task_status(self, task_id: str, status: TaskStatus) -> Optional[ExpertTask]:
        """更新任务状态 - Q65 适配方法

        替代原来 TaskQueue.update_status()
        """
        task = self.get_task(task_id)
        if task:
            task.status = status
            if status == TaskStatus.RUNNING:
                task.started_at = datetime.now()
            elif status in (TaskStatus.COMPLETED, TaskStatus.FAILED, TaskStatus.CANCELLED):
                task.completed_at = datetime.now()
            return self.update_task(task)
        return None

    def enqueue_task(self, task: ExpertTask) -> ExpertTask:
        """将任务状态设为 QUEUED - Q65 适配方法"""
        task.status = TaskStatus.QUEUED
        return self.create_task(task)
```

### 2.2 TaskScheduler（任务调度器）

```python
# task_scheduler.py
from typing import List, Dict, Callable, Optional
import asyncio
import time
from concurrent.futures import ThreadPoolExecutor, Future
import logging

logger = logging.getLogger(__name__)

class TaskScheduler:
    def __init__(self, max_concurrent: int = 4):
        self.max_concurrent = max_concurrent
        self._running: Dict[str, Future] = {}  # task_id -> Future
        self._executor = ThreadPoolExecutor(max_workers=max_concurrent)
        self._callbacks: Dict[str, Callable] = {}

    def submit(self, task_id: str, fn: Callable, *args, **kwargs) -> Future:
        """提交任务"""
        # 检查并发限制
        while len(self._running) >= self.max_concurrent:
            time.sleep(0.1)  # 等待

        future = self._executor.submit(fn, *args, **kwargs)
        self._running[task_id] = future

        # 添加完成回调
        def on_complete(f):
            self._running.pop(task_id, None)
            if task_id in self._callbacks:
                self._callbacks[task_id](f)

        future.add_done_callback(on_complete)
        return future

    def submit_async(self, task_id: str, coro):
        """提交异步任务"""
        loop = asyncio.get_event_loop()
        future = asyncio.run_coroutine_threadsafe(coro, loop)
        self._running[task_id] = future

        def on_complete(f):
            self._running.pop(task_id, None)

        future.add_done_callback(on_complete)
        return future

    def cancel(self, task_id: str) -> bool:
        """取消任务"""
        if task_id in self._running:
            cancelled = self._running[task_id].cancel()
            if cancelled:
                self._running.pop(task_id, None)
            return cancelled
        return False

    def wait_all(self, timeout: float = None) -> Dict[str, any]:
        """等待所有任务完成"""
        results = {}
        for task_id, future in self._running.items():
            try:
                results[task_id] = future.result(timeout=timeout)
            except Exception as e:
                results[task_id] = {"error": str(e)}
        return results

    def get_running_count(self) -> int:
        """获取运行中的任务数"""
        return len(self._running)

    def is_full(self) -> bool:
        """是否已满载"""
        return len(self._running) >= self.max_concurrent
```

### 2.3 TimeoutController（超时控制）

```python
# timeout_controller.py
import time
import threading
from typing import Dict, Callable, Optional, Any
from dataclasses import dataclass
import logging

logger = logging.getLogger(__name__)

@dataclass
class TimeoutConfig:
    default_timeout: float = 300  # 默认5分钟
    expert_timeouts: Dict[str, float] = None  # 专家特定超时

    def __post_init__(self):
        if self.expert_timeouts is None:
            self.expert_timeouts = {
                "kahneman": 180,
                "first_principle": 300,
                "six_hats": 240,
                "grow": 180,
                "reverse_thinking": 180,
                "default": 300
            }

    def get_timeout(self, expert_id: str) -> float:
        return self.expert_timeouts.get(expert_id, self.default_timeout)

class TimeoutController:
    def __init__(self, config: TimeoutConfig = None):
        self.config = config or TimeoutConfig()
        self._timers: Dict[str, threading.Timer] = {}
        self._callbacks: Dict[str, Callable] = {}
        self._task_start_times: Dict[str, float] = {}
        self._original_timeouts: Dict[str, float] = {}  # 保存原始超时值

    def start(self, task_id: str, expert_id: str, on_timeout: Callable):
        """启动超时计时器"""
        timeout = self.config.get_timeout(expert_id)
        self._callbacks[task_id] = on_timeout
        self._task_start_times[task_id] = time.time()
        self._original_timeouts[task_id] = timeout  # 保存原始超时值

        timer = threading.Timer(timeout, self._on_timeout, args=(task_id,))
        timer.daemon = True
        self._timers[task_id] = timer
        timer.start()

        logger.info(f"Started timeout timer for {task_id}: {timeout}s")

    def cancel(self, task_id: str):
        """取消超时计时"""
        if task_id in self._timers:
            self._timers[task_id].cancel()
            self._timers.pop(task_id, None)
            self._callbacks.pop(task_id, None)
            self._task_start_times.pop(task_id, None)
            self._original_timeouts.pop(task_id, None)

    def extend(self, task_id: str, extra_time: float):
        """延长超时时间"""
        if task_id in self._timers:
            # 取消当前计时器
            self._timers[task_id].cancel()

            # 计算剩余时间
            elapsed = time.time() - self._task_start_times.get(task_id, 0)
            # 使用保存的原始超时值计算剩余时间
            original_timeout = self._original_timeouts.get(task_id, 300)
            remaining = original_timeout - elapsed

            # 创建新的计时器
            new_timeout = max(remaining + extra_time, 60)  # 至少60秒
            timer = threading.Timer(new_timeout, self._on_timeout, args=(task_id,))
            timer.daemon = True
            self._timers[task_id] = timer
            # 更新保存的超时值
            self._original_timeouts[task_id] = new_timeout
            timer.start()

    def _on_timeout(self, task_id: str):
        """超时触发"""
        logger.warning(f"Task {task_id} timed out!")
        if task_id in self._callbacks:
            try:
                self._callbacks[task_id](task_id)
            except Exception as e:
                logger.error(f"Timeout callback error: {e}")

        # 清理
        self._timers.pop(task_id, None)
        self._callbacks.pop(task_id, None)

    def get_elapsed(self, task_id: str) -> float:
        """获取已用时间"""
        start = self._task_start_times.get(task_id)
        if start is None:
            return 0
        return time.time() - start

    def remaining(self, task_id: str) -> float:
        """获取剩余时间（秒），返回0表示已超时或任务不存在"""
        if task_id not in self._task_start_times:
            return 0
        elapsed = time.time() - self._task_start_times[task_id]
        original_timeout = self._original_timeouts.get(task_id, self.config.default_timeout)
        return max(original_timeout - elapsed, 0)
```

### 2.4 ResultCollector（结果收集器）

```python
# result_collector.py
from typing import Dict, List, Optional, Any
from dataclasses import dataclass, field
from enum import Enum
import time
import threading
import json
import logging

logger = logging.getLogger(__name__)

class ResultStatus(Enum):
    PENDING = "pending"
    RUNNING = "running"
    COMPLETED = "completed"
    FAILED = "failed"
    TIMEOUT = "timeout"
    CANCELLED = "cancelled"

@dataclass
class TaskResult:
    task_id: str
    expert_id: str
    status: ResultStatus = ResultStatus.PENDING
    output: Any = None
    error: Optional[str] = None
    started_at: Optional[float] = None
    completed_at: Optional[float] = None
    duration: Optional[float] = None
    tokens_used: int = 0
    metadata: Dict = field(default_factory=dict)

    def to_dict(self) -> dict:
        return {
            "task_id": self.task_id,
            "expert_id": self.expert_id,
            "status": self.status.value,
            "output": self.output,
            "error": self.error,
            "started_at": self.started_at,
            "completed_at": self.completed_at,
            "duration": self.duration,
            "tokens_used": self.tokens_used,
            "metadata": self.metadata
        }

class ResultCollector:
    def __init__(self, question_id: str, output_dir: str):
        self.question_id = question_id
        self.output_dir = output_dir
        self._results: Dict[str, TaskResult] = {}
        self._lock = threading.Lock()
        self._completed_event = threading.Event()

    def start(self, task_id: str, expert_id: str) -> TaskResult:
        """记录任务开始"""
        with self._lock:
            result = TaskResult(
                task_id=task_id,
                expert_id=expert_id,
                status=ResultStatus.RUNNING,
                started_at=time.time()
            )
            self._results[task_id] = result
            self._save_to_file(result)
            return result

    def complete(self, task_id: str, output: Any, tokens_used: int = 0, metadata: Dict = None):
        """记录任务完成"""
        with self._lock:
            if task_id not in self._results:
                logger.warning(f"Task {task_id} not found in results")
                return

            result = self._results[task_id]
            result.status = ResultStatus.COMPLETED
            result.output = output
            result.completed_at = time.time()
            result.duration = result.completed_at - result.started_at
            result.tokens_used = tokens_used
            if metadata:
                result.metadata.update(metadata)

            self._save_to_file(result)
            self._check_completion()

    def fail(self, task_id: str, error: str):
        """记录任务失败"""
        with self._lock:
            if task_id not in self._results:
                return

            result = self._results[task_id]
            result.status = ResultStatus.FAILED
            result.error = error
            result.completed_at = time.time()
            result.duration = result.completed_at - result.started_at

            self._save_to_file(result)
            self._check_completion()

    def timeout(self, task_id: str):
        """记录任务超时"""
        self.fail(task_id, "Task timeout")

    def get(self, task_id: str) -> Optional[TaskResult]:
        """获取结果"""
        return self._results.get(task_id)

    def get_all(self) -> List[TaskResult]:
        """获取所有结果"""
        return list(self._results.values())

    def get_completed(self) -> List[TaskResult]:
        """获取已完成的结果"""
        return [r for r in self._results.values() if r.status == ResultStatus.COMPLETED]

    def get_failed(self) -> List[TaskResult]:
        """获取失败的结果"""
        return [r for r in self._results.values() if r.status == ResultStatus.FAILED]

    def is_all_completed(self) -> bool:
        """是否全部完成"""
        if not self._results:
            return False
        return all(r.status == ResultStatus.COMPLETED for r in self._results.values())

    def wait_until_complete(self, timeout: float = None) -> bool:
        """等待全部完成"""
        return self._completed_event.wait(timeout)

    def _check_completion(self):
        """检查是否全部完成"""
        if self.is_all_completed():
            self._completed_event.set()

    def _save_to_file(self, result: TaskResult):
        """保存到文件"""
        import os
        os.makedirs(self.output_dir, exist_ok=True)
        file_path = os.path.join(self.output_dir, f"{result.task_id}.json")
        with open(file_path, 'w') as f:
            json.dump(result.to_dict(), f, ensure_ascii=False, indent=2)

    def export_summary(self) -> dict:
        """导出汇总"""
        completed = self.get_completed()
        failed = self.get_failed()

        return {
            "question_id": self.question_id,
            "total": len(self._results),
            "completed": len(completed),
            "failed": len(failed),
            "success_rate": len(completed) / len(self._results) if self._results else 0,
            "total_duration": sum(r.duration for r in completed if r.duration),
            "total_tokens": sum(r.tokens_used for r in completed),
            "results": [r.to_dict() for r in self._results.values()]
        }
```

### 3.1 ParallelExecutor（并行执行器）— 使用 Q64 TaskStore

```python
# parallel_executor.py
# Q65 使用 Q64 统一定义的 TaskStore，确保状态模型兼容
from typing import List, Dict, Callable, Optional
import logging
from concurrent.futures import ThreadPoolExecutor, as_completed

# 导入 Q64 统一定义的 TaskStore 和 ExpertTask
from task_store import TaskStore, ExpertTask, TaskStatus
from task_scheduler import TaskScheduler
from timeout_controller import TimeoutController, TimeoutConfig
from result_collector import ResultCollector, ResultStatus

logger = logging.getLogger(__name__)


class ParallelExecutor:
    def __init__(
        self,
        task_store: TaskStore,  # Q64 统一定义的 TaskStore
        max_concurrent: int = 4,
        timeout_config: TimeoutConfig = None,
        output_dir: str = "thinking-team/workspace/tasks"
    ):
        self.task_store = task_store  # 使用 Q64 的 TaskStore
        self.max_concurrent = max_concurrent
        self.timeout_config = timeout_config or TimeoutConfig()

        self.scheduler = TaskScheduler(max_concurrent=max_concurrent)
        self.timeout_controller = TimeoutController(self.timeout_config)

        self._running_tasks: Dict[str, str] = {}  # task_id -> question_id

    def execute_parallel(
        self,
        question_id: str,
        tasks: List[Dict],
        execute_fn: Callable,  # 执行函数: (expert_id, prompt) -> result
        on_task_complete: Callable = None
    ) -> Dict:
        """
        并行执行多个专家任务

        Args:
            question_id: 问题ID
            tasks: 任务列表 [{"expert_id": "kahneman", "prompt": "..."}, ...]
            execute_fn: 执行函数
            on_task_complete: 任务完成回调

        Returns:
            执行结果汇总

        Note:
            现在使用 Q64 的 TaskStore 统一管理任务状态，
            任务创建后状态为 QUEUED，执行时更新为 RUNNING，
            完成后更新为 COMPLETED 或 FAILED。
        """
        output_dir = f"thinking-team/workspace/tasks/{question_id}/results"
        collector = ResultCollector(question_id, output_dir)

        # 1. 通过 TaskStore 入队所有任务（使用 Q64 统一模型）
        for task in tasks:
            task_id = f"{question_id}_{task['expert_id']}"
            expert_task = ExpertTask(
                task_id=task_id,
                question_id=question_id,
                title=task.get("title", task["expert_id"]),
                description=task.get("description", ""),
                task_type=task.get("type", "analysis"),
                expert_id=task["expert_id"],
                prompt=task["prompt"],
                assigned_experts=[task["expert_id"]],
                status=TaskStatus.QUEUED  # 初始状态为 QUEUED
            )
            # 通过 TaskStore 创建任务（持久化到 JSON）
            self.task_store.create_task(expert_task)

        # 2. 并行执行
        with ThreadPoolExecutor(max_workers=self.max_concurrent) as executor:
            futures = {}

            for task in tasks:
                task_id = f"{question_id}_{task['expert_id']}"
                future = executor.submit(
                    self._execute_single_task,
                    task_id,
                    task["expert_id"],
                    task["prompt"],
                    collector,
                    execute_fn
                )
                futures[task_id] = future

            # 3. 等待完成
            for future in as_completed(futures):
                task_id = futures[future]
                try:
                    result = future.result()
                    if on_task_complete:
                        on_task_complete(task_id, result)
                except Exception as e:
                    logger.error(f"Task {task_id} failed: {e}")
                    collector.fail(task_id, str(e))

        # 4. 返回汇总
        return collector.export_summary()

    def _execute_single_task(
        self,
        task_id: str,
        expert_id: str,
        prompt: str,
        collector: ResultCollector,
        execute_fn: Callable
    ):
        """执行单个任务

        状态同步:
        1. 开始执行 → TaskStore 状态更新为 RUNNING
        2. 执行成功 → TaskStore 状态更新为 COMPLETED
        3. 执行失败 → TaskStore 状态更新为 FAILED
        """
        # 1. 通过 TaskStore 更新状态为 RUNNING（Q64 统一状态模型）
        task = self.task_store.get_task(task_id)
        if task:
            task.status = TaskStatus.RUNNING
            task.started_at = datetime.now()
            task.running_experts = [expert_id]
            self.task_store.update_task(task)

        # 2. 记录开始
        collector.start(task_id, expert_id)

        # 3. 启动超时计时器
        self.timeout_controller.start(
            task_id,
            expert_id,
            lambda tid: collector.timeout(tid)
        )

        try:
            # 4. 执行
            result = execute_fn(expert_id, prompt)

            # 5. 记录完成
            collector.complete(
                task_id,
                output=result.get("output"),
                tokens_used=result.get("tokens_used", 0),
                metadata=result.get("metadata", {})
            )

            # 6. 通过 TaskStore 更新状态为 COMPLETED
            task = self.task_store.get_task(task_id)
            if task:
                task.status = TaskStatus.COMPLETED
                task.completed_at = datetime.now()
                task.completed_experts = [expert_id]
                task.output = result.get("output", {})
                self.task_store.update_task(task)

            # 7. 取消超时计时器
            self.timeout_controller.cancel(task_id)

            return result

        except Exception as e:
            logger.error(f"Task {task_id} error: {e}")
            # 通过 TaskStore 更新状态为 FAILED
            task = self.task_store.get_task(task_id)
            if task:
                task.status = TaskStatus.FAILED
                task.completed_at = datetime.now()
                task.errors.append(str(e))
                self.task_store.update_task(task)
            collector.fail(task_id, str(e))
            self.timeout_controller.cancel(task_id)
            raise

    def execute_chain(
        self,
        question_id: str,
        tasks: List[Dict],
        execute_fn: Callable,
        combine_fn: Callable = None
    ) -> Dict:
        """
        链式执行（上一个的输出作为下一个的输入）

        使用 Q64 TaskStore 统一管理任务状态。

        Args:
            question_id: 问题ID
            tasks: 任务列表（按顺序）
            execute_fn: 执行函数
            combine_fn: 组合函数（可选）

        Returns:
            执行结果
        """
        output_dir = f"thinking-team/workspace/tasks/{question_id}/results"
        collector = ResultCollector(question_id, output_dir)

        context = {}

        for i, task in enumerate(tasks):
            task_id = f"{question_id}_chain_{i}"

            # 如果有前一个任务的输出，合并到prompt
            if context and "context" in task:
                task["prompt"] = combine_fn(
                    task["prompt"],
                    context
                ) if combine_fn else task["prompt"]

            # 通过 TaskStore 创建任务
            expert_task = ExpertTask(
                task_id=task_id,
                question_id=question_id,
                title=task.get("title", f"Chain step {i}"),
                description=task.get("description", ""),
                task_type="chain",
                expert_id=task["expert_id"],
                prompt=task["prompt"],
                assigned_experts=[task["expert_id"]],
                status=TaskStatus.QUEUED
            )
            self.task_store.create_task(expert_task)

            # 执行
            collector.start(task_id, task["expert_id"])

            # 更新状态为 RUNNING
            self.task_store.update_task_status(task_id, TaskStatus.RUNNING)

            try:
                result = execute_fn(task["expert_id"], task["prompt"])

                collector.complete(
                    task_id,
                    output=result.get("output"),
                    tokens_used=result.get("tokens_used", 0)
                )

                # 更新状态为 COMPLETED
                self.task_store.update_task_status(task_id, TaskStatus.COMPLETED)

                # 保存上下文
                context[task["expert_id"]] = result.get("output")

            except Exception as e:
                logger.error(f"Chain task {task_id} failed: {e}")
                # 更新状态为 FAILED
                self.task_store.update_task_status(task_id, TaskStatus.FAILED)
                collector.fail(task_id, str(e))
                break

        return collector.export_summary()

    def execute_debate(
        self,
        question_id: str,
        tasks: List[Dict],
        execute_fn: Callable,
        max_rounds: int = 3
    ) -> Dict:
        """
        辩论模式执行（多轮交互）

        使用 Q64 TaskStore 统一管理任务状态。

        Args:
            question_id: 问题ID
            tasks: 初始任务列表
            execute_fn: 执行函数
            max_rounds: 最大轮数

        Returns:
            辩论结果
        """
        output_dir = f"thinking-team/workspace/tasks/{question_id}/results"
        collector = ResultCollector(question_id, output_dir)

        # 第一轮：并行执行初始分析
        expert_outputs = {}
        for task in tasks:
            task_id = f"{question_id}_round0_{task['expert_id']}"

            # 通过 TaskStore 创建任务
            expert_task = ExpertTask(
                task_id=task_id,
                question_id=question_id,
                title=f"Debate Round 0 - {task['expert_id']}",
                description="Initial debate analysis",
                task_type="debate",
                expert_id=task["expert_id"],
                prompt=task["prompt"],
                assigned_experts=[task["expert_id"]],
                status=TaskStatus.QUEUED
            )
            self.task_store.create_task(expert_task)

            collector.start(task_id, task["expert_id"])
            self.task_store.update_task_status(task_id, TaskStatus.RUNNING)

            try:
                result = execute_fn(task["expert_id"], task["prompt"])
                collector.complete(task_id, output=result.get("output"))
                self.task_store.update_task_status(task_id, TaskStatus.COMPLETED)
                expert_outputs[task["expert_id"]] = result.get("output")
            except Exception as e:
                collector.fail(task_id, str(e))
                self.task_store.update_task_status(task_id, TaskStatus.FAILED)

        # 后续轮：基于他人输出进行辩论
        for round_num in range(1, max_rounds):
            new_outputs = {}

            for task in tasks:
                # 构建包含他人观点的prompt
                debate_prompt = self._build_debate_prompt(
                    task["prompt"],
                    expert_outputs,
                    task["expert_id"]
                )

                task_id = f"{question_id}_round{round_num}_{task['expert_id']}"

                # 通过 TaskStore 创建任务
                expert_task = ExpertTask(
                    task_id=task_id,
                    question_id=question_id,
                    title=f"Debate Round {round_num} - {task['expert_id']}",
                    description=f"Debate round {round_num}",
                    task_type="debate",
                    expert_id=task["expert_id"],
                    prompt=debate_prompt,
                    assigned_experts=[task["expert_id"]],
                    status=TaskStatus.QUEUED
                )
                self.task_store.create_task(expert_task)

                collector.start(task_id, task["expert_id"])
                self.task_store.update_task_status(task_id, TaskStatus.RUNNING)

                try:
                    result = execute_fn(task["expert_id"], debate_prompt)
                    collector.complete(task_id, output=result.get("output"))
                    self.task_store.update_task_status(task_id, TaskStatus.COMPLETED)
                    new_outputs[task["expert_id"]] = result.get("output")
                except Exception as e:
                    collector.fail(task_id, str(e))
                    self.task_store.update_task_status(task_id, TaskStatus.FAILED)

            # 检查是否收敛（所有专家达成共识）
            if self._check_convergence(new_outputs, expert_outputs):
                logger.info(f"Debate converged at round {round_num}")
                break

            expert_outputs = new_outputs

        return collector.export_summary()

    def _build_debate_prompt(self, original_prompt: str, other_outputs: Dict, my_expert_id: str) -> str:
        """构建辩论prompt"""
        others_view = "\n\n".join([
            f"### {expert_id}的观点:\n{output}"
            for expert_id, output in other_outputs.items()
            if expert_id != my_expert_id
        ])

        return f"""
原始问题: {original_prompt}

其他专家的观点:
{others_view}

请基于以上观点，提出你的反驳或补充意见。
"""

    def _check_convergence(self, new_outputs: Dict, old_outputs: Dict) -> bool:
        """检查是否收敛

        P1修正: 修复维度错误 - 原来是 changed <= len(...) * 0.3
        (count vs count×percentage 比较无意义)
        修正为: 变化比例 = changed / len(...) <= 0.3
        """

        if not old_outputs:
            return False

        # 统计变化的专家数量
        changed = 0
        total = len(new_outputs)

        for expert_id in new_outputs:
            if expert_id in old_outputs:
                if new_outputs[expert_id] != old_outputs[expert_id]:
                    changed += 1

        # P1修正: 比较变化比例 vs 阈值
        if total == 0:
            return True

        change_ratio = changed / total
        return change_ratio <= 0.3
```

## 六、总结

### 6.1 核心组件

| 组件 | 功能 | 关键类 | 来源 |
|------|------|--------|------|
| TaskStore | 任务存储持久化 | create_task, get_task, update_task, list_tasks | **Q64统一** |
| TaskScheduler | 并发控制 | submit, cancel, wait_all | Q65 |
| TimeoutController | 超时管理 | start, cancel, extend, get_elapsed, remaining | Q65 |
| ResultCollector | 结果收集 | start, complete, fail, export_summary | Q65 |
| ParallelExecutor | 执行引擎 | execute_parallel, execute_chain, execute_debate | Q65 |

### 6.2 执行模式

| 模式 | 适用场景 | 方法 |
|------|----------|------|
| **并行** | 独立分析，多角度 | execute_parallel |
| **链式** | 依赖分析，顺序推理 | execute_chain |
| **辩论** | 观点碰撞，达成共识 | execute_debate |

### 6.3 Q64/Q65 状态模型统一（P0问题C3修复）

**问题**: Q64/Q65 任务状态模型完全不兼容

**解决方案**: Q65 使用 Q64 统一定义的 TaskStore

| 统一点 | Q64 定义 | Q65 使用方式 |
|--------|----------|--------------|
| 状态枚举 | `TaskStatus` (CREATED, QUEUED, RUNNING, COMPLETED, FAILED等) | 直接导入使用 |
| 任务模型 | `ExpertTask` Pydantic模型 | 直接导入使用 |
| 存储接口 | `TaskStore` (JSON文件持久化) | 构造时注入 |
| 状态流转 | Q64 定义的标准流转 | Q65 通过 `update_task_status()` 更新 |

**状态流转图**:
```
CREATED → QUEUED → RUNNING → COMPLETED
                    ↓
                  FAILED / CANCELLED
```

**使用示例**:
```python
from task_store import TaskStore, ExpertTask, TaskStatus

# 初始化 TaskStore
task_store = TaskStore("thinking-team/workspace/tasks")

# 创建执行器（注入统一 TaskStore）
executor = ParallelExecutor(task_store=task_store, max_concurrent=4)

# 执行时自动通过 TaskStore 同步状态
result = executor.execute_parallel(
    question_id="q_001",
    tasks=[{"expert_id": "kahneman", "prompt": "..."}],
    execute_fn=call_expert
)

# 可通过 TaskStore 查询任务状态
task = task_store.get_task("q_001_kahneman")
print(f"Task status: {task.status}")  # COMPLETED
```
