## 一、设计目标与原则

### 1.1 核心目标

| 目标 | 说明 |
|------|------|
| **瓶颈识别** | 快速定位哪个环节最耗时 |
| **根因分析** | 深入分析瓶颈产生的根本原因 |
| **量化评估** | 建立"最耗时"的量化标准 |
| **基线对比** | 建立性能基线，支持历史对比 |
| **持续优化** | 追踪优化效果，形成闭环 |

### 1.2 设计原则

| 原则 | 说明 |
|------|------|
| **分层监控** | 与Q26监控指标体系保持一致（L1-L4四层） |
| **数据驱动** | 基于Q20文件体系采集性能数据 |
| **实时可见** | 瓶颈信息实时可查 |
| **可操作性** | 瓶颈定位后有明确的优化方向 |

## 四、"最耗时"量化标准

### 4.1 量化维度

定义三个维度量化"最耗时"：

| 维度 | 说明 | 计算方式 |
|------|------|---------|
| **绝对耗时** | 环节实际花费的时间 | 直接计时 |
| **占比** | 环节耗时占总耗时比例 | 环节耗时 / 总耗时 |
| **偏离度** | 相对于基线的偏离程度 | (实际耗时 - 基线耗时) / 基线耗时 |

### 4.2 瓶颈判定规则

```python
# 瓶颈判定算法
def identify_bottleneck(task_metrics: dict) -> dict:
    """
    识别任务中的性能瓶颈
    返回: 瓶颈环节、耗时详情、优化建议
    """

    total_duration = task_metrics["total_duration_seconds"]

    # 计算各环节耗时和占比
   环节_metrics = []
    for phase in ["intent", "distribute", "expert", "message", "debate", "sync"]:
        phase_duration = task_metrics[f"{phase}_duration_seconds"]
        phase_ratio = phase_duration / total_duration
        phase_baseline = baselines[phase]
        phase_deviation = (phase_duration - phase_baseline) / phase_baseline

        环节_metrics.append({
            "phase": phase,
            "duration": phase_duration,
            "ratio": phase_ratio,
            "baseline": phase_baseline,
            "deviation": phase_deviation
        })

    # 按占比排序
    环节_metrics.sort(key=lambda x: x["ratio"], reverse=True)

    # 瓶颈判定
    bottleneck = 环节_metrics[0]

    # 判定规则
    if bottleneck["ratio"] > 0.5:
        level = "critical"  # 占比超过50%，严重瓶颈
    elif bottleneck["ratio"] > 0.3:
        level = "major"     # 占比30%-50%，主要瓶颈
    elif bottleneck["deviation"] > 0.5:
        level = "anomaly"   # 偏离度超过50%，异常瓶颈
    else:
        level = "normal"    # 正常范围

    return {
        "bottleneck": bottleneck,
        "all_phases": 环节_metrics,
        "level": level,
        "recommendation": get_recommendation(bottleneck)
    }
```

### 4.3 瓶颈级别定义

| 级别 | 条件 | 颜色标识 | 行动 |
|------|------|---------|------|
| **Critical** | 占比 > 50% | 🔴 红色 | 立即优化 |
| **Major** | 占比 30%-50% | 🟠 橙色 | 优先优化 |
| **Minor** | 占比 15%-30% | 🟡 黄色 | 关注 |
| **Normal** | 占比 < 15% | 🟢 绿色 | 无需行动 |

### 4.4 专家级瓶颈判定

```python
# 专家级瓶颈判定
def identify_expert_bottleneck(expert_metrics: dict) -> dict:
    """
    识别专家执行中的性能瓶颈
    """

    # 分解专家耗时
    load_time = expert_metrics["load_context_seconds"]
    llm_time = expert_metrics["llm_call_seconds"]
    output_time = expert_metrics["output_generation_seconds"]
    wait_time = expert_metrics["wait_seconds"]
    sync_time = expert_metrics.get("state_sync_seconds", 0)

    total = load_time + llm_time + output_time + wait_time + sync_time

    # 计算占比
    breakdown = {
        "load_context": {"time": load_time, "ratio": load_time / total},
        "llm_call": {"time": llm_time, "ratio": llm_time / total},
        "output_generation": {"time": output_time, "ratio": output_time / total},
        "wait": {"time": wait_time, "ratio": wait_time / total},
        "state_sync": {"time": sync_time, "ratio": sync_time / total}
    }

    # 找出最大占比
    bottleneck_phase = max(breakdown.items(), key=lambda x: x[1]["ratio"])

    # 对比基线
    phase_baseline = expert_baselines[expert_metrics["expert_id"]][bottleneck_phase[0]]
    deviation = (bottleneck_phase[1]["time"] - phase_baseline) / phase_baseline

    return {
        "expert_id": expert_metrics["expert_id"],
        "total_duration": total,
        "breakdown": breakdown,
        "bottleneck_phase": bottleneck_phase[0],
        "bottleneck_time": bottleneck_phase[1]["time"],
        "baseline": phase_baseline,
        "deviation": deviation,
        "is_anomaly": deviation > 0.5
    }
```

### 5.3 基线更新机制

```python
# 基线自动更新脚本
class BaselineUpdater:
    """性能基线自动更新"""

    def __init__(self, base_path: str):
        self.base_path = Path(base_path)
        self.monitoring_path = self.base_path / "monitoring"

    def update_baselines(self, days: int = 7):
        """
        基于过去N天的数据更新基线
        默认每周更新一次
        """

        # 收集过去N天的任务数据
        task_data = self._collect_task_data(days)

        # 更新环节基线
        self._update_phase_baselines(task_data)

        # 更新专家基线
        self._update_expert_baselines(task_data)

        # 更新场景基线
        self._update_scenario_baselines(task_data)

        # 记录更新日志
        self._log_update(days, len(task_data))

    def _collect_task_data(self, days: int) -> list:
        """收集过去N天的任务数据"""
        tasks = []
        cutoff = datetime.now() - timedelta(days=days)

        # 从workspace/tasks收集
        tasks_path = self.base_path / "workspace" / "tasks"
        for question_dir in tasks_path.iterdir():
            if not question_dir.is_dir():
                continue

            for task_file in (question_dir / "tasks").glob("*.json"):
                task = json.loads(task_file.read_text())
                if datetime.fromisoformat(task["metadata"]["completed_at"]) > cutoff:
                    tasks.append(task)

        return tasks

    def _calculate_percentiles(self, values: list) -> dict:
        """计算分位数"""
        if not values:
            return {"p50": 0, "p90": 0, "p95": 0}

        import numpy as np
        return {
            "p50": round(np.percentile(values, 50), 2),
            "p90": round(np.percentile(values, 90), 2),
            "p95": round(np.percentile(values, 95), 2)
        }

    def _update_phase_baselines(self, tasks: list):
        """更新环节基线"""
        phases = {
            "intent": [],
            "distribute": [],
            "expert": [],
            "message": [],
            "debate": [],
            "sync": []
        }

        for task in tasks:
            for phase in phases.keys():
                if f"{phase}_duration_seconds" in task.get("metadata", {}):
                    phases[phase].append(
                        task["metadata"][f"{phase}_duration_seconds"]
                    )

        baselines = {
            "version": datetime.now().strftime("%Y-%m-%d"),
            "last_updated": datetime.now().isoformat(),
            "baselines": {}
        }

        for phase, values in phases.items():
            baselines["baselines"][phase] = self._calculate_percentiles(values)

        # 保存
        file_path = self.monitoring_path / "baselines" / "phase_baselines.json"
        file_path.parent.mkdir(parents=True, exist_ok=True)
        file_path.write_text(json.dumps(baselines, indent=2, ensure_ascii=False))
```

```python
    """获取性能趋势"""
    analyzer = PerformanceAnalyzer(BASE_PATH)
    return analyzer.get_trends(days)


@router.websocket("/ws/performance")
async def websocket_performance(websocket: WebSocket):
    """WebSocket实时推送性能数据"""
    await websocket.accept()

    analyzer = PerformanceAnalyzer(BASE_PATH)

    while True:
        # 获取所有运行中的任务
        running_tasks = analyzer.get_running_tasks()

        bottlenecks = []
        for task in running_tasks:
            bottleneck = analyzer.identify_bottleneck(task["question_id"])
            bottlenecks.append({
                "question_id": task["question_id"],
                "bottleneck": bottleneck
            })

        await websocket.send_json({
            "timestamp": datetime.now().isoformat(),
            "tasks": bottlenecks
        })

        await asyncio.sleep(10)  # 每10秒推送
```

## 八、性能分析引擎实现

### 8.1 核心类设计

```python
# scripts/analyzer.py
import json
from pathlib import Path
from datetime import datetime, timedelta
from typing import Dict, List, Optional
import numpy as np


class PerformanceAnalyzer:
    """性能瓶颈分析引擎"""

    def __init__(self, base_path: str):
        self.base_path = Path(base_path)
        self.workspace_path = self.base_path / "workspace"
        self.performance_path = self.base_path / "performance"
        self.baselines = self._load_baselines()

    def _load_baselines(self) -> dict:
        """加载性能基线"""
        baseline_file = self.performance_path / "baselines" / "phase_baselines.json"
        if baseline_file.exists():
            return json.loads(baseline_file.read_text())
        return {}

    def identify_bottleneck(self, question_id: str) -> dict:
        """识别任务瓶颈"""
        # 加载任务数据
        task_data = self._load_task_data(question_id)

        if not task_data:
            return {"error": "Task not found"}

        # 分解各环节耗时
        phases = self._decompose_phases(task_data)

        # 计算总耗时
        total = sum(p["duration"] for p in phases.values())

        # 计算占比
        for phase in phases.values():
            phase["ratio"] = phase["duration"] / total if total > 0 else 0

        # 按占比排序
        sorted_phases = sorted(phases.items(), key=lambda x: x[1]["ratio"], reverse=True)

        # 确定瓶颈
        bottleneck = sorted_phases[0]
        bottleneck_level = self._get_bottleneck_level(bottleneck[1]["ratio"])

        return {
            "question_id": question_id,
            "total_duration": total,
            "phases": phases,
            "bottleneck": {
                "phase": bottleneck[0],
                "duration": bottleneck[1]["duration"],
                "ratio": bottleneck[1]["ratio"],
                "level": bottleneck_level
            },
            "expert_ranking": self._get_expert_ranking(task_data)
        }

    def _decompose_phases(self, task_data: dict) -> dict:
        """分解各环节耗时"""
        phases = {}

        # 从metadata提取各环节耗时
        metadata = task_data.get("metadata", {})

        # 意图分析
        phases["intent"] = {
            "duration": metadata.get("intent_duration_seconds", 2),
            "baseline": self.baselines.get("baselines", {}).get("intent", {}).get("p50", 3)
        }

        # 任务分发
        phases["distribute"] = {
            "duration": metadata.get("distribute_duration_seconds", 1),
            "baseline": self.baselines.get("baselines", {}).get("distribute", {}).get("p50", 1)
        }

        # 专家分析（汇总所有专家）
        expert_total = 0
        for expert in task_data.get("experts", []):
            if expert.get("status") == "completed":
                exp_meta = expert.get("metadata", {})
                expert_total += exp_meta.get("duration_seconds", 0)

        phases["expert"] = {
            "duration": expert_total,
            "baseline": self.baselines.get("baselines", {}).get("expert", {}).get("p50", 120)
        }

        # 消息传递
        phases["message"] = {
            "duration": metadata.get("message_duration_seconds", 2),
            "baseline": self.baselines.get("baselines", {}).get("message", {}).get("p50", 2)
        }

        # 辩论整合
        phases["debate"] = {
            "duration": metadata.get("debate_duration_seconds", 30),
            "baseline": self.baselines.get("baselines", {}).get("debate", {}).get("p50", 30)
        }

        # 状态同步
        phases["sync"] = {
            "duration": metadata.get("sync_duration_seconds", 1),
            "baseline": self.baselines.get("baselines", {}).get("sync", {}).get("p50", 1)
        }

        return phases

    def _get_bottleneck_level(self, ratio: float) -> str:
        """获取瓶颈级别"""
        if ratio > 0.5:
            return "critical"
        elif ratio > 0.3:
            return "major"
        elif ratio > 0.15:
            return "minor"
        else:
            return "normal"

    def _get_expert_ranking(self, task_data: dict) -> List[dict]:
        """获取专家耗时排名"""
        experts = []

        for expert in task_data.get("experts", []):
            if expert.get("status") == "completed":
                meta = expert.get("metadata", {})
                duration = meta.get("duration_seconds", 0)

                # 分解专家耗时
                perf = meta.get("performance", {})
                experts.append({
                    "expert_id": expert["id"],
                    "total_duration": duration,
                    "llm_call": perf.get("llm_call_seconds", 0),
                    "load_context": perf.get("load_context_seconds", 0),
                    "output_generation": perf.get("output_generation_seconds", 0),
                    "wait": perf.get("wait_seconds", 0)
                })

        # 按耗时排序
        return sorted(experts, key=lambda x: x["total_duration"], reverse=True)

    def compare_with_baseline(self, question_id: str) -> dict:
        """与基线对比"""
        bottleneck = self.identify_bottleneck(question_id)

        comparison = {
            "total": {
                "actual": bottleneck["total_duration"],
                "baseline_p50": sum(p["baseline"] for p in bottleneck["phases"].values()),
                "deviation": 0
            },
            "phases": {}
        }

        baseline_total = comparison["total"]["baseline_p50"]
        if baseline_total > 0:
            comparison["total"]["deviation"] = (
                (bottleneck["total_duration"] - baseline_total) / baseline_total * 100
            )

        for phase, data in bottleneck["phases"].items():
            baseline = data["baseline"]
            deviation = ((data["duration"] - baseline) / baseline * 100) if baseline > 0 else 0

            comparison["phases"][phase] = {
                "actual": data["duration"],
                "baseline": baseline,
                "deviation": deviation,
                "status": "anomaly" if deviation > 50 else "warning" if deviation > 20 else "normal"
            }

        return comparison

    def get_trends(self, days: int = 7) -> dict:
        """获取性能趋势"""
        trends = {
            "period": f"过去{days}天",
            "data_points": [],
            "summary": {}
        }

        # 收集历史数据
        cutoff = datetime.now() - timedelta(days=days)
        analysis_path = self.performance_path / "analysis"

        if not analysis_path.exists():
            return trends

        for task_dir in analysis_path.iterdir():
            if not task_dir.is_dir():
                continue

            try:
                # 从目录名解析日期
                question_id = task_dir.name
                # 简化处理：假设按question_id排序即可
                analysis_file = task_dir / "bottleneck_analysis.json"

                if analysis_file.exists():
                    data = json.loads(analysis_file.read_text())
                    trends["data_points"].append({
                        "question_id": question_id,
                        "total_duration": data.get("total_duration", 0),
                        "bottleneck": data.get("bottleneck", {}).get("phase", "unknown")
                    })
            except Exception:
                continue

        # 计算汇总
        if trends["data_points"]:
            durations = [d["total_duration"] for d in trends["data_points"]]
            trends["summary"] = {
                "avg_duration": np.mean(durations),
                "p90_duration": np.percentile(durations, 90),
                "total_tasks": len(durations)
            }

        return trends
```
