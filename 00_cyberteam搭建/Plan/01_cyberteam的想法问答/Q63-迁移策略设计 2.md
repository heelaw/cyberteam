# 思考天团迁移策略机制设计

**问题**: 如何设计思考天团的数据迁移机制，确保版本升级时的数据兼容与平滑过渡？

---

## 一、迁移策略设计原则

### 1.1 设计目标

| 目标 | 说明 |
|------|------|
| **数据完整性** | 迁移过程中不丢失任何有效数据 |
| **平滑过渡** | 用户无感知切换，不影响使用体验 |
| **可回滚** | 迁移失败时可快速恢复到迁移前状态 |
| **可观测** | 迁移过程可监控、可追踪、可审计 |
| **自动化** | 减少人工干预，降低出错概率 |

### 1.2 与灰度升级的关系

迁移策略是灰度升级（Q62）的数据层保障：

```
┌─────────────────────────────────────────────────────────────────────────┐
│                     灰度升级 + 迁移策略关系                               │
├─────────────────────────────────────────────────────────────────────────┤
│                                                                          │
│  灰度升级 (Q62)                                                         │
│  ├── 流量分发: 0% → 5% → 30% → 70% → 100%                            │
│  ├── 状态机: 运行中 / 暂停 / 回滚                                       │
│  └── 决策引擎: 指标驱动升级                                             │

│                                                                          │
└─────────────────────────────────────────────────────────────────────────┘
```

## 三、迁移策略模式

### 3.1 策略模式对比

| 策略 | 适用场景 | 优点 | 缺点 | 风险等级 |
|------|---------|------|------|---------|
| **一次性迁移** | 数据量小、离线场景 | 简单直接 | 停机时间长 | 中 |
| **渐进式迁移** | 大数据量、在线场景 | 风险可控 | 实现复杂 | 低 |
| **并行迁移** | 关键业务、高可用 | 零停机 | 资源消耗大 | 低 |
| **双写策略** | 持续迭代场景 | 平滑过渡 | 逻辑复杂 | 中 |

### 4.1 兼容策略矩阵

| 场景 | 策略 | 实现方式 |
|------|------|---------|
| 新增字段 | 默认值填充 | 配置默认值或从上下文推断 |

### 4.2 字段映射处理

```python
# migration/field_mapping.py

from typing import Dict, Any, Callable, Optional

class FieldMapper:
    """字段映射器"""

    def __init__(self):
        self.mapping_rules: Dict[str, Dict[str, Any]] = {}

    def register_rule(
        self,
        source_field: str,
        target_field: str,
        transform: Optional[Callable] = None,
        default: Any = None
    ):
        """注册映射规则"""
        self.mapping_rules[source_field] = {
            "target": target_field,
            "transform": transform,
            "default": default
        }

    def map(self, source_data: Dict[str, Any]) -> Dict[str, Any]:
        """执行映射"""
        result = {}

        for source_field, rule in self.mapping_rules.items():
            if source_field in source_data:
                value = source_data[source_field]

                # 应用转换函数
                if rule["transform"]:
                    value = rule["transform"](value)

                result[rule["target"]] = value

            elif rule["default"] is not None:
                # 使用默认值
                result[rule["target"]] = rule["default"]

        return result


# 示例：v1.0 → v2.0 字段映射
mapper = FieldMapper()

# 新增字段：使用默认值
mapper.register_rule(
    source_field="",
    target_field="columns",
    default=[{"id": "default", "name": "默认列"}]
)

# 字段重命名：使用转换函数
mapper.register_rule(
    source_field="created_at",
    target_field="created_timestamp",
    transform=lambda x: int(x.timestamp() * 1000)
)

# 字段类型转换
mapper.register_rule(
    source_field="is_active",
    target_field="status",
    transform=lambda x: "enabled" if x else "disabled"
)
```

```python
# migration/defaults.py

class DefaultValueProvider:
    """默认值提供者"""

    def __init__(self):
        self.defaults = {
            # 显式默认值
            "columns": [{"id": "default", "name": "默认列"}],
            "settings": {
                "theme": "light",
                "language": "zh-CN"
            },

            # 系统默认值
            "interest_tags": ["未设置"],
            "priority": 0,
            "is_enabled": True,

            # 空值标记
            "deleted_at": "__MIGRATION_PENDING__"
        }

    def get_default(self, field: str, context: Dict = None) -> Any:
        """获取默认值"""
        if field in self.defaults:
            return self.defaults[field]

        # 尝试推断
        if context and field == "columns":
            return self._infer_columns(context)

        return None

    def _infer_columns(self, context: Dict) -> List[Dict]:
        """从上下文推断列信息"""
        cards = context.get("cards", [])
        if not cards:
            return [{"id": "default", "name": "默认列"}]

        # 从卡片中提取列信息
        columns = []
        for card in cards:
            if "column" in card:
                col_id = card["column"]
                if col_id not in [c["id"] for c in columns]:
                    columns.append({"id": col_id, "name": col_id})

        return columns if columns else [{"id": "default", "name": "默认列"}]
```

### 4.4 版本兼容性处理

```python
# migration/version_compat.py

from typing import Dict, Any, List
import semver  # 语义化版本

class VersionCompatManager:
    """版本兼容性管理器"""

    def __init__(self):
        self.compat_rules: Dict[str, Dict] = {}

    def register_compat_rule(
        self,
        from_version: str,
        to_version: str,
        migrator: callable
    ):
        """注册兼容规则"""
        key = f"{from_version}->{to_version}"
        self.compat_rules[key] = {
            "from": from_version,
            "to": to_version,
            "migrator": migrator
        }

    def get_migration_path(self, from_ver: str, to_ver: str) -> List[str]:
        """获取迁移路径"""
        # 简单版本：直接迁移
        if f"{from_ver}->{to_ver}" in self.compat_rules:
            return [f"{from_ver}->{to_ver}"]

        # 复杂版本：逐步升级
        # v1.0 → v1.5 → v2.0
        path = []
        current = from_ver

        while current != to_ver:
            next_ver = self._find_next_version(current, to_ver)
            path.append(f"{current}->{next_ver}")
            current = next_ver

        return path

    def migrate(self, data: Dict[str, Any], from_ver: str, to_ver: str) -> Dict[str, Any]:
        """执行迁移"""
        path = self.get_migration_path(from_ver, to_ver)

        for step in path:
            rule = self.compat_rules[step]
            data = rule["migrator"](data)

        return data


# 示例：注册迁移规则
compat_manager = VersionCompatManager()

# v1.0 → v1.5 迁移
compat_manager.register_compat_rule("v1.0", "v1.5", lambda data: {
    **data,
    "columns": data.get("columns", [{"id": "default", "name": "默认列"}])
})

# v1.5 → v2.0 迁移
compat_manager.register_compat_rule("v1.5", "v2.0", lambda data: {
    **data,
    "settings": {
        "theme": data.get("theme", "light"),
        "language": data.get("language", "zh-CN")
    }
})
```

## 五、回滚机制设计

### 5.1 回滚策略

| 回滚类型 | 触发条件 | 回滚速度 | 数据影响 |
|----------|---------|---------|---------|
| **快速回滚** | P0级错误 | 秒级 | 仅配置 |
| **标准回滚** | P1级错误 | 分钟级 | 配置+部分数据 |
| **完整回滚** | 严重数据错误 | 小时级 | 全部数据 |

### 5.2 回滚流程设计

```
┌─────────────────────────────────────────────────────────────────────────┐
```

```python
# migration/snapshot.py

import json
import os
import shutil
from datetime import datetime
from typing import Dict, Any, Optional

class SnapshotManager:
    """快照管理器"""

    def __init__(self, snapshot_dir: str = "backups/snapshots"):
        self.snapshot_dir = snapshot_dir
        os.makedirs(snapshot_dir, exist_ok=True)

    def create_snapshot(self, data: Dict[str, Any], tag: str = "") -> str:
        """创建快照"""
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        snapshot_id = f"snapshot_{timestamp}_{tag}" if tag else f"snapshot_{timestamp}"

        snapshot_path = os.path.join(self.snapshot_dir, snapshot_id)
        os.makedirs(snapshot_path, exist_ok=True)

        # 写入数据
        for key, value in data.items():
            file_path = os.path.join(snapshot_path, f"{key}.json")
            with open(file_path, "w", encoding="utf-8") as f:
                json.dump(value, f, ensure_ascii=False, indent=2)

        # 写入元数据
        metadata = {
            "snapshot_id": snapshot_id,
            "created_at": timestamp,
            "tags": [tag] if tag else []
        }
        with open(os.path.join(snapshot_path, "metadata.json"), "w") as f:
            json.dump(metadata, f, indent=2)

        return snapshot_id

    def restore_snapshot(self, snapshot_id: str) -> Dict[str, Any]:
        """恢复快照"""
        snapshot_path = os.path.join(self.snapshot_dir, snapshot_id)

        if not os.path.exists(snapshot_path):
            raise ValueError(f"Snapshot {snapshot_id} not found")

        data = {}
        for filename in os.listdir(snapshot_path):
            if filename.endswith(".json") and filename != "metadata.json":
                key = filename.replace(".json", "")
                with open(os.path.join(snapshot_path, filename), "r") as f:
                    data[key] = json.load(f)

        return data

    def list_snapshots(self, tag: Optional[str] = None) -> list:
        """列出快照"""
        snapshots = []
        for dirname in os.listdir(self.snapshot_dir):
            metadata_path = os.path.join(self.snapshot_dir, dirname, "metadata.json")
            if os.path.exists(metadata_path):
                with open(metadata_path, "r") as f:
                    metadata = json.load(f)

                if tag is None or tag in metadata.get("tags", []):
                    snapshots.append(metadata)

        return sorted(snapshots, key=lambda x: x["created_at"], reverse=True)

    def delete_snapshot(self, snapshot_id: str):
        """删除快照"""
        snapshot_path = os.path.join(self.snapshot_dir, snapshot_id)
        if os.path.exists(snapshot_path):
            shutil.rmtree(snapshot_path)
```

```python
# migration/scripts/board_v1_to_v2.py

def migrate_board_v1_to_v2(board_v1: dict) -> dict:
    """
    将 board v1.0 迁移到 v2.0

    v1.0: { id, title, cards: [{id, content, created_at}] }
    v2.0: { id, title, cards, columns, settings }
    """

    # 1. 保留原有字段
    board_v2 = {
        "id": board_v1["id"],
        "title": board_v1["title"],
        "cards": board_v1["cards"]
    }

    # 2. 新增 columns 字段 (从 cards 推断)
    columns = []
    for card in board_v1.get("cards", []):
        column_name = "待办"  # 默认列
        # 从卡片属性推断列
        if "column" in card:
            column_name = card["column"]

        if column_name not in [c["name"] for c in columns]:
            columns.append({
                "id": column_name.lower().replace(" ", "_"),
                "name": column_name
            })

    # 如果没有列，默认创建一个
    if not columns:
        columns = [{"id": "default", "name": "默认列"}]

    board_v2["columns"] = columns

    # 3. 新增 settings 字段 (默认值)
    board_v2["settings"] = {
        "theme": "light",
        "language": "zh-CN",
        "card_sort": "created_at",
        "auto_save": True
    }

    # 4. 迁移标记
    board_v2["_migration_info"] = {
        "from_version": "v1.0",
        "to_version": "v2.0",
        "migrated_at": "2024-01-01T00:00:00Z"
    }

    return board_v2
```

```python
# migration/integration.py

class MigrationGrayscaleIntegration:
    """迁移与灰度集成"""

    def __init__(self):
        self.grayscale_config = None
        self.migration_config = None

    def make_integrated_decision(self, metrics: dict) -> dict:
        """
        综合决策：考虑灰度指标 + 迁移指标

        返回:
        {
            "action": "promote/pause/rollback",
            "reasons": [...],
            "migration_status": {...},
            "grayscale_status": {...}
        }
        """
        grayscale_decision = self._get_grayscale_decision(metrics)
        migration_decision = self._get_migration_decision(metrics)

        # 优先级：迁移问题 > 灰度问题
        if migration_decision.action == "rollback":
            return {
                "action": "rollback",
                "reasons": migration_decision.reasons,
                "migration_status": migration_decision,
                "grayscale_status": grayscale_decision
            }

        if grayscale_decision.action == "rollback":
            return {
                "action": "rollback",
                "reasons": grayscale_decision.reasons,
                "migration_status": migration_decision,
                "grayscale_status": grayscale_decision
            }

        # 继续灰度
        return {
            "action": grayscale_decision.action,
            "reasons": grayscale_decision.reasons + migration_decision.reasons,
            "migration_status": migration_decision,
            "grayscale_status": grayscale_decision
        }

    def _get_migration_decision(self, metrics: dict) -> dict:
        """获取迁移决策"""
        # 检查迁移指标
        if metrics.get("migration_error_rate", 0) > 0.01:
            return {"action": "rollback", "reasons": ["迁移错误率过高"]}

        if metrics.get("data_consistency_rate", 1) < 0.99:
            return {"action": "pause", "reasons": ["数据一致性不足"]}

        return {"action": "continue", "reasons": []}
```
