```python
# timeout_controller.py

class TimeoutController:
    def __init__(self, config: TimeoutConfig = None):
        self.config = config or TimeoutConfig()
        self._timers: Dict[str, threading.Timer] = {}
        self._callbacks: Dict[str, Callable] = {}
        self._task_start_times: Dict[str, float] = {}
        self._original_timeouts: Dict[str, float] = {}
        self._complexity_multiplier: Dict[str, float] = {}  # 新增: 复杂度倍数

    def start(
        self,
        task_id: str,
        expert_id: str,
        on_timeout: Callable,
        complexity_multiplier: float = 1.0  # 新增参数
    ):
        """启动超时计时器

        Args:
            task_id: 任务ID
            expert_id: 专家ID
            on_timeout: 超时回调
            complexity_multiplier: 复杂度倍数 (来自Q68 complexity_result.to_execution_params())
        """
        base_timeout = self.config.get_timeout(expert_id)
        # 根据复杂度调整超时
        timeout = base_timeout * complexity_multiplier

        self._callbacks[task_id] = on_timeout
        self._task_start_times[task_id] = time.time()
        self._original_timeouts[task_id] = timeout
        self._complexity_multiplier[task_id] = complexity_multiplier

        timer = threading.Timer(timeout, self._on_timeout, args=(task_id,))
        timer.daemon = True
        self._timers[task_id] = timer
        timer.start()

        logger.info(f"Started timeout timer for {task_id}: {timeout}s (multiplier={complexity_multiplier})")
```

## 三、统一收敛检查逻辑

### 3.1 问题

三套实现各自为政：
- Q65: 简单字符串比较
- Q67: Jaccard相似度 (正确但重复)
- Q68: 评分法 (用于迭代改进判断，非辩论收敛)

### 3.2 解决方案: 统一ConvergenceChecker接口

```python
# convergence_checker.py
from abc import ABC, abstractmethod
from typing import Dict, List, Set
import re

class ConvergenceChecker(ABC):
    """收敛检查器抽象接口"""

    @abstractmethod
    def check(
        self,
        current_outputs: Dict[str, str],
        previous_outputs: Dict[str, str]
    ) -> bool:
        """
        检查是否收敛

        Args:
            current_outputs: 当前轮次输出 {expert_id: output_text}
            previous_outputs: 上一轮输出

        Returns:
            True if converged (可以停止辩论), False otherwise
        """
        pass

    @abstractmethod
    def get_name(self) -> str:
        """检查器名称"""
        pass


class StandardConvergenceChecker(ConvergenceChecker):
    """
    标准收敛检查器 (Q65/Q67统一版)

    收敛条件: 变化比例 <= threshold
    - 使用Jaccard相似度计算语义变化
    - 变化比例 = 变化的专家数 / 总专家数
    """

    def __init__(self, threshold: float = 0.3):
        self.threshold = threshold

    def check(
        self,
        current_outputs: Dict[str, str],
        previous_outputs: Dict[str, str]
    ) -> bool:
        if not previous_outputs:
            return False

        total = len(current_outputs)
        if total == 0:
            return True

        changed = 0
        for expert_id, output in current_outputs.items():
            prev_output = previous_outputs.get(expert_id, "")
            if not self._is_stable(output, prev_output):
                changed += 1

        change_ratio = changed / total
        return change_ratio <= self.threshold

    def _is_stable(self, current: str, previous: str) -> bool:
        """判断输出是否稳定 (使用Jaccard相似度)"""
        if not previous:
            return False

        curr_keywords = self._extract_keywords(current)
        prev_keywords = self._extract_keywords(previous)

        if not curr_keywords or not prev_keywords:
            return current == previous

        intersection = len(curr_keywords & prev_keywords)
        union = len(curr_keywords | prev_keywords)

        if union == 0:
            return True

        similarity = intersection / union
        # 相似度 >= 0.7 认为稳定
        return similarity >= 0.7

    def _extract_keywords(self, text: str) -> Set[str]:
        """提取关键词"""
        if not text:
            return set()

        stopwords = {
            "的", "了", "是", "在", "和", "与", "对", "为", "有",
            "the", "a", "an", "is", "are", "and", "or", "to", "of"
        }

        words = re.findall(r'[\w\u4e00-\u9fff]{2,}', text.lower())
        return {w for w in words if w not in stopwords and len(w) > 1}

    def get_name(self) -> str:
        return f"StandardConvergenceChecker(threshold={self.threshold})"


class ImprovementConvergenceChecker(ConvergenceChecker):
    """
    改进收敛检查器 (基于Q68评分法)

    用于迭代改进判断，非辩论收敛
    收敛条件: 改进度 < min_improvement
    """

    def __init__(self, min_improvement: float = 0.1):
        self.min_improvement = min_improvement

    def check(
        self,
        current_outputs: Dict[str, str],
        previous_outputs: Dict[str, str]
    ) -> bool:
        if not previous_outputs:
            return False

        improvement = self._calculate_improvement(current_outputs, previous_outputs)
        return improvement < self.min_improvement

    def _calculate_improvement(
        self,
        current: Dict[str, str],
        previous: Dict[str, str]
    ) -> float:
        """计算改进度"""
        current_score = self._score_outputs(current)
        previous_score = self._score_outputs(previous)

        if previous_score == 0:
            return 1.0

        return (current_score - previous_score) / previous_score

    def _score_outputs(self, outputs: Dict[str, str]) -> float:
        """评分输出 (来自Q68)"""
        score = 0.0

        for output in outputs.values():
            if not output:
                continue

            # 长度分数
            score += min(len(output) / 100, 30)

            # 结构分数
            if "##" in output:
                score += 20

            # 建议分数
            if "建议" in output or "方案" in output:
                score += 20

            # 风险分数
            if "风险" in output or "问题" in output:
                score += 15

        return score

    def get_name(self) -> str:
        return f"ImprovementConvergenceChecker(min={self.min_improvement})"
```
