# ========================================
# 复盘流程
# ========================================
review_workflow:
  step_1_keep:
    name: "识别保持项"
    actions:
      - "回顾成功经验"
      - "识别有效做法"
      - "分析成功原因"

    output: "保持清单"

  step_2_improve:
    name: "识别改进项"
    actions:
      - "找出不足之处"
      - "分析改进空间"
      - "提出改进方案"

    output: "改进清单"

  step_3_start:
    name: "识别开始项"
    actions:
      - "发现新机会"
      - "提出新尝试"
      - "规划新做法"

    output: "开始清单"

  step_4_stop:
    name: "识别停止项"
    actions:
      - "识别无效做法"
      - "发现时间浪费"
      - "找出阻碍因素"

    output: "停止清单"

knowledge_base:
  theory: |
    KISS复盘的要点：

    1. Keep (保持):
       - 识别成功经验
       - 巩固有效做法
       - 形成最佳实践

    2. Improve (改进):
       - 找出不足
       - 优化流程
       - 提升效率

    3. Start (开始):
       - 发现新机会
       - 尝试新方法
       - 持续创新

    4. Stop (停止):
       - 及时止损
       - 减少浪费
       - 聚焦价值

    复盘原则：
    - 客观真实
    - 不指责个人
    - 关注系统性
    - 持续改进
