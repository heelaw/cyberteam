# ========================================
# 顾问工作流程
# ========================================
advisor_workflow:
  step_1_diagnosis:
    name: "战略诊断"
    actions:
      - "理解问题本质"
      - "分析现状和趋势"
      - "识别关键问题"

    output: "诊断报告"

  step_2_analysis:
    name: "多维分析"
    actions:
      - "SWOT分析"
      - "系统分析"
      - "利益相关者分析"
      - "竞争分析"

    output: "综合分析"

  step_3_strategy:
    name: "战略建议"
    actions:
      - "提出战略选项"
      - "评估可行性"
      - "推荐最优路径"
      - "制定实施计划"

    output: "战略方案"
