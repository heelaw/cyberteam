# ========================================
# 分析流程
# ========================================
analysis_workflow:
  step_1_internal:
    name: "内部分析(SW)"
    actions:
      - "识别内部优势"
      - "识别内部劣势"
      - "评估核心竞争力"

    output: "SW分析"

  step_2_external:
    name: "外部分析(OT)"
    actions:
      - "识别外部机会"
      - "识别外部威胁"
      - "评估环境趋势"

    output: "OT分析"

  step_3_cross:
    name: "交叉分析"
    actions:
      - "SO: 发挥优势抓住机会"
      - "WO: 克服劣势抓住机会"
      - "ST: 发挥优势应对威胁"
      - "WT: 减少劣势规避威胁"

    output: "战略矩阵"

knowledge_base:
  theory: |
    SWOT分析的要点：

    1. 内部分析(SW):
       - 优势: 擅长、独特、资源
       - 劣势: 不足、缺失、弱点

    2. 外部分析(OT):
       - 机会: 趋势、新可能、有利变化
       - 威胁: 竞争、风险、不利变化

    3. 战略组合:
       - SO: 增长型战略(最佳)
       - WO: 扭转性战略(改进)
       - ST: 多元化战略(防御)
       - WT: 防御性战略(保底)

    应用要点：
    - 客观真实
    - 具体明确
    - 分清内外
    - 关注交叉
