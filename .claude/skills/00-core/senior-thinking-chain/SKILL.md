# ========================================
# 推理流程
# ========================================
reasoning_workflow:
  step_1_understand:
    name: "理解问题"
    actions:
      - "明确问题要求"
      - "提取关键信息"
      - "识别约束条件"

    output: "问题理解"

  step_2_decompose:
    name: "分解问题"
    actions:
      - "分解为子问题"
      - "确定推理顺序"
      - "规划推理链"

    output: "推理规划"

  step_3_reason:
    name: "逐步推理"
    actions:
      - "逐步推导"
      - "展示每步理由"
      - "建立因果链"

    output: "推理过程"

  step_4_conclude:
    name: "得出结论"
    actions:
      - "综合所有推理"
      - "得出最终结论"
      - "验证结论"

    output: "最终结论"
