## 方法一：使用浏览器控制台脚本（推荐）

### 步骤：

1. **访问课程页面**
   - 打开浏览器，访问：https://www.sanjieke.cn/path_detail/1189
   - 登录三节课账号

2. **进入课程学习页面**
   - 找到目标课程并点击
   - 等待页面加载完成

3. **打开浏览器开发者工具**
   - 按 F12 或右键 → 检查
   - 切换到 Console（控制台）标签

4. **复制并粘贴以下脚本**

```javascript
// 三节课逐字稿提取脚本
(async function extractAllTranscripts() {
    const transcripts = [];

    // 提取当前章节逐字稿
    function extractCurrent() {
        const allP = Array.from(document.querySelectorAll('p'));
        let startIdx = -1;
        let endIdx = allP.length;

        allP.forEach((p, idx) => {
            const text = p.textContent.trim();
            if (text === '同学你好' || text.includes('课程导入')) {
                startIdx = idx;
            }
            if ((text.includes('大家都在问') || text.includes('向讲师提问')) && startIdx !== -1) {
                endIdx = idx;
            }
        });

        let content = [];
        if (startIdx !== -1) {
            for (let i = startIdx; i < endIdx; i++) {
                const text = allP[i].textContent.trim();
                if (text) content.push(text);
            }
        }

        // 获取章节标题
        const titleEl = document.querySelector('[class*="outline"] li[class*="active"] p') ||
                         document.querySelector('h1, h2');
        let title = titleEl ? titleEl.textContent.trim() : '未知章节';

        return {
            title: title,
            content: content.join('\n\n')
        };
    }

    // 点击下一节
    function clickNext() {
        const links = Array.from(document.querySelectorAll('a'));
        const nextLink = links.find(l => l.textContent.includes('下一节'));
        if (nextLink) {
            nextLink.click();
            return true;
        }
        return false;
    }

    // 等待函数
    const wait = (ms) => new Promise(resolve => setTimeout(resolve, ms));

    // 提取所有章节（最多13章）
    for (let i = 0; i < 13; i++) {
        console.log(`正在提取第 ${i + 1} 章...`);

        const transcript = extractCurrent();
        transcripts.push(transcript);

        if (i < 12) {
            clickNext();
            await wait(3000);
        }
    }

    // 下载为文件
    const markdown = transcripts.map((t, i) =>
        `## 第${i+1}章: ${t.title}\n\n${t.content}\n\n---\n\n`
    ).join('\n');

    const blob = new Blob([markdown], { type: 'text/markdown;charset=utf-8' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = '课程逐字稿.md';
    a.click();

    console.log('提取完成！');
    return transcripts;
})();
```

5. **运行脚本**
   - 粘贴脚本后按回车
   - 等待脚本执行完成
   - 文件将自动下载
