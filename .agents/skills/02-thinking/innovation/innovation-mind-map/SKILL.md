# ========================================
# 制作流程
# ========================================
creation_workflow:
  step_1_center:
    name: "中心主题"
    actions:
      - "确定核心问题/主题"
      - "放在中心位置"
      - "用图像或醒目文字"

    output: "中心图像"

  step_2_branches:
    name: "主干分支"
    actions:
      - "从中心延伸主要分支"
      - "每条分支代表一个维度"
      - "使用不同颜色区分"

    output: "主要分支"

  step_3_details:
    name: "细节展开"
    actions:
      - "每个分支延伸细节"
      - "使用关键词而非句子"
      - "添加图标和图像"

    output: "完整导图"

# 知识库
# ========================================
knowledge_base:
  theory: |
    思维导图的要素：

    1. 中心主题：核心焦点
    2. 主分支：主要分类（建议3-7个）
    3. 子分支：细节展开
    4. 关键词：每点一个核心词
    5. 图像：增强记忆
    6. 颜色：区分分支
    7. 连线：表示关联

    制作技巧：
    - 从中心开始，向外发散
    - 使用曲线而非直线
    - 每条分支一个关键词
    - 使用图像和颜色
    - 保持层次清晰
