    definition: "多个模型共同支持的结论"

    value: "核心结论的基础"

  unique_insight:
    definition: "单个模型的独特视角"

    definition: "模型间的分歧点"

  step_1_collect:
    name: "收集洞察"
    actions:
      - "从每个模型输出中提取关键观点"
      - "标记洞察类型"
      - "初步分类"

    output: "原始洞察清单"

  step_2_integrate:
    name: "整合洞察"
    actions:
      - "识别共识点"
      - "整合相似观点"
      - "处理分歧"

    output: "整合后的洞察"

  step_3_structure:
    name: "结构化"
    actions:
      - "建立洞察间关系"
      - "构建因果链"
      - "识别杠杆点"

    output: "结构化洞察体系"
