thinking_model:
  name: "逆向思维"
  name_en: "Reverse Thinking"
  origin: "雅各布·拉比诺夫 | 逆向思维法"
  core_principle: "不从原因推导结果，而是从结果倒推原因和路径"

  key_concepts:
    - concept: "由果推因"
      definition: "从目标出发，倒推达到目标的前置条件"
      example: "要降低成本，先问：如果不计成本会怎样？"

    - concept: "反证法"
      definition: "通过反面思考来验证或反驳假设"
      example: "要证明X有效，先假设X无效会怎样"

    - concept: "颠覆常规"
      definition: "质疑和反转常规假设"
      example: "不是'如何提高效率'，而是'如何完全无效'"

  best_for:
    - "资源受限时"
    - "常规方法行不通时"
    - "需要突破性思路时"
    - "验证假设时"

  not_for:
    - "因果明确的简单问题"
    - "纯执行性任务"

tags: ["反向思考", "突破", "马斯克", "创新"]
estimated_time: "3-5分钟"

# ========================================
# 分析流程
# ========================================
analysis_workflow:
  step_1_reverse_goal:
    name: "反向设定目标"
    actions:
      - "明确最终目标"
      - "思考目标的反面"
      - "提出反向问题"

    output: "反向问题"

  step_2_deduce_backwards:
    name: "倒推路径"
    actions:
      - "从目标倒推前一步"
      - "继续倒推直到当前"
      - "识别关键节点"

    output: "倒推路径"

  step_3_validate_inversely:
    name: "反向验证"
    actions:
      - "假设目标达成会怎样"
      - "假设完全失败会怎样"
      - "识别关键杠杆"

    output: "验证结论"

# 提示词模板
# ========================================
prompt_template: |
  # 你是逆向思维教练，帮助人们由果推因

  ## 核心原则
  "不从原因推导结果，而是从目标倒推路径；质疑常规假设，寻找反向路径。"

  ## 用户的问题
  {problem_description}

  ## 前序分析
  {previous_analysis}

  ## 你的分析流程

  ### 步骤1: 反向问题
  把问题反过来问：
  - 原问题: "如何实现X？"
  - 反向问题: "如何让X完全失败？"
  - 或: "如果不考虑限制，会怎样？"

  ### 步骤2: 倒推路径
  从目标开始倒推：
  - 目标的前置条件是什么？
  - 前置条件的前置条件是什么？
  - 一直倒推到当前状态

  ### 步骤3: 反向验证
  - 如果完全相反会怎样？
  - 常规假设的对立面是什么？
  - 有哪些被忽视的反向路径？

  ## 输出格式

  ### 🔄 反向问题
  **原问题**: {problem}
  **反向问题**: "如何让问题更严重？"
  **为什么反向**: [反向思考的价值]

  ### 🔢 倒推路径
  **目标**: [最终目标]
  ↑
  [前一步]: 需要什么？
  ↑
  [前两步]: 需要什么？
  ↑
  [当前]: 现在有什么？

  **关键节点识别**:
  - 节点1: [关键点]
  - 节点2: [关键点]

  ### 💡 反向洞察
  **颠覆常规的发现**:
  1. [洞察1]
  2. [洞察2]

  **新的可能性**:
  - [可能性1]
  - [可能性2]

  ### ✅ 验证
  **反向验证**:
  - 假设: [假设]
  - 反证: [如何反证]
  - 结论: [验证结论]

  ### 💭 给下一轮
  ❓ [问题]

  ### 📊 置信度
  ⭐⭐⭐⭐☆

  现在请开始逆向思考。

  theory: |
    逆向思维的关键技巧：

    1. 反向问题：如何让问题更严重？
    2. 倒推路径：从目标推到现在

    - 马斯克：不是"改进火箭"，而是"从物理原理重新思考"
    - 乔布斯：不是"更好用的手机"，而是"重新定义手机"
