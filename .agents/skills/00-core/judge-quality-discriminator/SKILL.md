    name: "收集输出"

      - "整理分析内容"
      - "提取关键观点"

    output: "输出汇总"

  step_2_evaluate:
    name: "评估质量"
    actions:
      - "逐个评估每个输出"
      - "多维度评分"
      - "识别亮点和问题"

    output: "质量评估"

  step_3_discriminate:
    name: "判别分类"
    actions:
      - "区分高质量/低质量"
      - "识别突出模型"
      - "标记问题输出"

    output: "判别结果"
